## Key Architecture Notes

- **Test runner:** Jest (NOT Vitest) - backend uses `packages/backend/jest.config.js`
- **Mock interfaces:** `IClaudeClient` + `IVisionClient` (NOT concrete `ClaudeClient`)
- **Assert on:** `messages[0].content` passed to `sendMessage(messages, options)`
- **Service-populated fields:** `rawTextExcerpt`, `sourceFilePath` (NOT from Claude response)
- **Vision path:** Images route through `visionClient.prepareDocument` + `analyzeImages`

```typescript
// packages/backend/__tests__/unit/DocumentParserService.test.ts

const mockPdfParse = jest.fn();
jest.mock('pdf-parse', () => ({ __esModule: true, default: mockPdfParse }));

const mockMammothExtractRawText = jest.fn();
jest.mock('mammoth', () => ({
  __esModule: true,
  default: { extractRawText: mockMammothExtractRawText },
}));

import { DocumentParserService } from '../../src/infrastructure/ai/DocumentParserService.js';
import type { IClaudeClient, ClaudeMessage } from '../../src/application/interfaces/IClaudeClient.js';
import type { IVisionClient } from '../../src/application/interfaces/IVisionClient.js';
import type { DocumentMetadata, DocumentType } from '../../src/application/interfaces/IDocumentParser.js';
import { AssessmentNotFoundError } from '../../src/application/interfaces/IScoringDocumentParser.js';

function createMetadata(overrides?: Partial<DocumentMetadata>): DocumentMetadata {
  return {
    filename: 'test.pdf',
    mimeType: 'application/pdf',
    sizeBytes: 1024,
    documentType: 'pdf' as DocumentType,
    storagePath: '/tmp/uploads/test.pdf',
    uploadedAt: new Date(),
    uploadedBy: 'user-123',
    ...overrides,
  };
}

describe('DocumentParserService (unit)', () => {
  let service: DocumentParserService;
  let mockClaude: jest.Mocked<IClaudeClient>;
  let mockVision: jest.Mocked<IVisionClient>;

  beforeEach(() => {
    jest.clearAllMocks();

    mockClaude = {
      sendMessage: jest.fn(),
      streamMessage: jest.fn() as any,
    };

    mockVision = {
      prepareDocument: jest.fn(),
      analyzeImages: jest.fn(),
    } as any;

    service = new DocumentParserService(mockClaude, mockVision);

    mockPdfParse.mockResolvedValue({ text: 'Mock PDF content', numpages: 1 });
    mockMammothExtractRawText.mockResolvedValue({ value: 'Mock DOCX content' });
  });

  describe('parseForContext (intake)', () => {
    it('uses pdf-parse text and calls claudeClient.sendMessage with DOCUMENT CONTENT', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: JSON.stringify({
          vendorName: 'Acme Health',
          solutionName: 'DiagnosticsAI',
          solutionType: 'Clinical Decision Support',
          industry: 'Healthcare',
          features: ['A', 'B'],
          claims: [],
          integrations: [],
          complianceMentions: ['HIPAA'],
          architectureNotes: [],
          securityMentions: [],
          confidence: 0.9,
          suggestedQuestions: ['Q1'],
          coveredCategories: ['security_risk'],
          gapCategories: ['privacy_risk'],
        }),
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForContext(Buffer.from('pdf-bytes'), createMetadata());

      expect(result.success).toBe(true);
      expect(mockPdfParse).toHaveBeenCalled();

      expect(mockClaude.sendMessage).toHaveBeenCalled();
      const [messages] = mockClaude.sendMessage.mock.calls[0] as [ClaudeMessage[]];
      expect(messages[0].content).toContain('DOCUMENT CONTENT:\nMock PDF content');

      // Service-populated fields (not Claude)
      expect(result.context?.rawTextExcerpt).toBe('Mock PDF content');
      expect(result.context?.sourceFilePath).toBe('/tmp/uploads/test.pdf');

      expect(result.suggestedQuestions).toEqual(['Q1']);
      expect(result.gapCategories).toEqual(['privacy_risk']);
    });

    it('uses mammoth text for docx', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: JSON.stringify({
          vendorName: 'Docx Vendor',
          solutionName: null,
          solutionType: null,
          industry: null,
          features: [],
          claims: [],
          integrations: [],
          complianceMentions: [],
          architectureNotes: [],
          securityMentions: [],
          confidence: 0.7,
          suggestedQuestions: [],
          coveredCategories: [],
          gapCategories: [],
        }),
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForContext(
        Buffer.from('docx-bytes'),
        createMetadata({
          filename: 'test.docx',
          mimeType: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
          documentType: 'docx',
        })
      );

      expect(result.success).toBe(true);
      expect(mockMammothExtractRawText).toHaveBeenCalled();
      const [messages] = mockClaude.sendMessage.mock.calls[0] as [ClaudeMessage[]];
      expect(messages[0].content).toContain('DOCUMENT CONTENT:\nMock DOCX content');
    });

    it('routes images through visionClient (no claudeClient.sendMessage)', async () => {
      mockVision.prepareDocument.mockResolvedValue([{ type: 'base64', data: 'abc', mediaType: 'image/png' }] as any);
      mockVision.analyzeImages.mockResolvedValue({
        content: JSON.stringify({
          vendorName: 'Image Vendor',
          solutionName: null,
          solutionType: null,
          industry: null,
          features: [],
          claims: [],
          integrations: [],
          complianceMentions: [],
          architectureNotes: [],
          securityMentions: [],
          confidence: 0.8,
          suggestedQuestions: [],
          coveredCategories: [],
          gapCategories: [],
        }),
      } as any);

      const result = await service.parseForContext(
        Buffer.from('png-bytes'),
        createMetadata({ filename: 'img.png', mimeType: 'image/png', documentType: 'image' })
      );

      expect(result.success).toBe(true);
      expect(mockVision.prepareDocument).toHaveBeenCalled();
      expect(mockVision.analyzeImages).toHaveBeenCalled();
      expect(mockClaude.sendMessage).not.toHaveBeenCalled();
    });

    it('handles markdown fenced JSON', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: '```json\n{"vendorName":"FenceCo","confidence":0.8,"features":[],"claims":[],"integrations":[],"complianceMentions":[],"architectureNotes":[],"securityMentions":[],"suggestedQuestions":[],"coveredCategories":[],"gapCategories":[],"solutionName":null,"solutionType":null,"industry":null}\n```',
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForContext(Buffer.from('pdf'), createMetadata());
      expect(result.success).toBe(true);
      expect(result.context?.vendorName).toBe('FenceCo');
    });

    it('fails gracefully on invalid JSON', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: 'not json',
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForContext(Buffer.from('pdf'), createMetadata());
      expect(result.success).toBe(false);
      expect(result.context).toBeNull();
      expect(result.error).toBeDefined();
    });
  });

  describe('parseForResponses (scoring)', () => {
    it('returns responses on success', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: JSON.stringify({
          assessmentId: 'assess-123',
          vendorName: 'Acme',
          solutionName: 'Tool',
          responses: [
            {
              sectionNumber: 1,
              sectionTitle: 'Privacy',
              questionNumber: 1,
              questionText: 'Q?',
              responseText: 'A',
              confidence: 0.95,
              hasVisualContent: false,
              visualContentDescription: null,
            },
          ],
          expectedQuestionCount: 90,
          parsedQuestionCount: 1,
          unparsedQuestions: [],
          isComplete: false,
          overallConfidence: 0.9,
        }),
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForResponses(Buffer.from('pdf'), createMetadata());
      expect(result.success).toBe(true);
      expect(result.assessmentId).toBe('assess-123');
      expect(result.responses).toHaveLength(1);
    });

    it('throws AssessmentNotFoundError when assessmentId missing', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: JSON.stringify({
          assessmentId: null,
          vendorName: null,
          solutionName: null,
          responses: [],
          expectedQuestionCount: null,
          parsedQuestionCount: 0,
          unparsedQuestions: [],
          isComplete: false,
          overallConfidence: 0.4,
        }),
        stop_reason: 'end_turn',
        model: 'test',
      });

      await expect(service.parseForResponses(Buffer.from('pdf'), createMetadata())).rejects.toThrow(
        AssessmentNotFoundError
      );
    });

    it('filters low-confidence responses when minConfidence set (includeLowConfidence=false)', async () => {
      mockClaude.sendMessage.mockResolvedValue({
        content: JSON.stringify({
          assessmentId: 'assess-123',
          vendorName: null,
          solutionName: null,
          responses: [
            { sectionNumber: 1, questionNumber: 1, questionText: 'Q1', responseText: 'A1', confidence: 0.9, hasVisualContent: false, visualContentDescription: null },
            { sectionNumber: 1, questionNumber: 2, questionText: 'Q2', responseText: 'A2', confidence: 0.5, hasVisualContent: false, visualContentDescription: null },
          ],
          expectedQuestionCount: null,
          parsedQuestionCount: 2,
          unparsedQuestions: [],
          isComplete: false,
          overallConfidence: 0.7,
        }),
        stop_reason: 'end_turn',
        model: 'test',
      });

      const result = await service.parseForResponses(Buffer.from('pdf'), createMetadata(), {
        minConfidence: 0.7,
        includeLowConfidence: false,
      });

      expect(result.responses).toHaveLength(1);
      expect(result.responses[0].confidence).toBe(0.9);
    });
  });
});
```
