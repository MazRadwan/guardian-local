## Implementation

### E2E Test File

**Path:** `apps/web/e2e/intake-upload.spec.ts`

```typescript
// apps/web/e2e/intake-upload.spec.ts
import { test, expect } from '@playwright/test';

/**
 * Quick login helper - uses existing dev quick login button
 * Matches pattern from other E2E tests in this repo
 */
async function devQuickLogin(page: import('@playwright/test').Page) {
  await page.goto('/login');
  await page.getByRole('button', { name: 'Quick Login (test@guardian.com)' }).click();
  await expect(page.getByRole('textbox', { name: 'Message input' })).toBeVisible({ timeout: 10000 });
}

test.describe('Epic 16: intake upload (smoke)', () => {
  test('uploads a document and shows extracted context message', async ({ page }) => {
    // Clear any stale state
    await page.goto('/');
    await page.evaluate(() => localStorage.clear());

    // Login via quick login button
    await devQuickLogin(page);

    // Ensure a conversation exists (upload is disabled without conversationId)
    await page.getByRole('button', { name: 'New Chat' }).click();

    // Upload a small "pdf-like" payload (validation should accept header)
    const fileChooserPromise = page.waitForEvent('filechooser');
    await page.getByRole('button', { name: 'Attach file' }).click();
    const chooser = await fileChooserPromise;

    await chooser.setFiles({
      name: 'vendor.pdf',
      mimeType: 'application/pdf',
      buffer: Buffer.from('%PDF-1.4\n%Mock\n'),
    });

    // UI should enter progress state quickly (frontend optimistic state)
    await expect(page.getByText(/Uploading file|Processing upload|Analyzing document/i)).toBeVisible({
      timeout: 5000,
    });

    // Backend should emit a chat message with the standardized header
    // This is deterministic - the formatIntakeContextMessage always includes this text
    await expect(page.getByText('Document Context Extracted')).toBeVisible({
      timeout: 30000,
    });
  });
});
```

## Environment Requirements

### Required Environment Variables

For E2E tests, environment variables must be loaded when processes start. Use one of these approaches:

**Option A: Add to existing `.env` files (recommended)**

| Variable | Value | File | Purpose |
|----------|-------|------|---------|
| `NEXT_PUBLIC_ENABLE_DEV_MODE` | `true` | `apps/web/.env.local` | Shows "Quick Login" button on login page |
| `MOCK_DOCUMENT_PARSER` | `true` | `packages/backend/.env` | Uses stub parser instead of Claude API |

**Option B: Export when launching processes**

```bash
# Terminal 1: Backend with mock parser
cd packages/backend && MOCK_DOCUMENT_PARSER=true pnpm dev

# Terminal 2: Frontend with dev mode
cd apps/web && NEXT_PUBLIC_ENABLE_DEV_MODE=true pnpm dev

# Terminal 3: Run E2E
cd apps/web && pnpm exec playwright test intake-upload
```

**Option C: Playwright config (frontend only)**

```typescript
// apps/web/playwright.config.ts
export default defineConfig({
  use: {
    // ...
  },
  webServer: {
    command: 'NEXT_PUBLIC_ENABLE_DEV_MODE=true pnpm dev',
    // ...
  },
});
```

**Note:** `NEXT_PUBLIC_*` vars must be set at build/start time (not runtime) for Next.js client-side access.
