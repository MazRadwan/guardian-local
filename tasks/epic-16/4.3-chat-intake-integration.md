## Implementation

### Architecture Decision: Where to Save Context Message

**Key insight:** Intake parsing happens in `DocumentUploadController` (Story 2.1), not ChatServer.
The controller already has the parsed context and emits `intake_context_ready`.
Therefore, the "save assistant message" logic must also live in the controller.

**IMPORTANT:** Must use `assistant` role, NOT `system` role.
`ChatServer.buildConversationContext` filters to only `user`/`assistant` messages,
so system messages won't reach Claude. Using `assistant` ensures the context
appears in the last-10 message history that Claude sees.

### Backend: Update DocumentUploadController

**Path:** `packages/backend/src/infrastructure/http/controllers/DocumentUploadController.ts`

Update the `parseForIntake` method (from Story 2.1) to save the assistant message:

```typescript
// Add ConversationService to constructor dependencies:
constructor(
  private readonly fileStorage: IFileStorage,
  private readonly fileValidator: FileValidationService,
  private readonly intakeParser: IIntakeDocumentParser,
  private readonly scoringParser: IScoringDocumentParser,
  private readonly conversationRepository: IConversationRepository,
  private readonly conversationService: ConversationService,  // ADD THIS
  private readonly chatNamespace: Namespace
) {}

/**
 * Parse document for intake context
 *
 * When parsing succeeds:
 * 1. Save context as ASSISTANT message (so Claude sees it)
 * 2. Emit 'message' event (so it appears in chat)
 * 3. Emit 'intake_context_ready' (for UI state updates)
 */
private async parseForIntake(
  socketRoom: string,
  buffer: Buffer,
  metadata: DocumentMetadata,
  conversationId: string,
  uploadId: string
): Promise<void> {
  const result = await this.intakeParser.parseForContext(buffer, metadata, {
    conversationId,
  });

  if (result.success && result.context) {
    // 1. Format context as assistant message
    // Note: gapCategories is on IntakeParseResult, not IntakeContext (per Story 1.2)
    const contextMessage = this.formatIntakeContextMessage(result.context, result.gapCategories);

    // 2. Save as ASSISTANT message (not system!) so Claude sees it
    // ChatServer.buildConversationContext filters to user/assistant only
    const savedMessage = await this.conversationService.sendMessage({
      conversationId,
      role: 'assistant',  // CRITICAL: 'assistant', NOT 'system'
      content: { text: contextMessage },
    });

    // 3. Emit 'message' event so it appears in chat UI
    this.chatNamespace.to(socketRoom).emit('message', {
      id: savedMessage.id,
      conversationId,
      role: 'assistant',
      content: savedMessage.content,
      createdAt: savedMessage.createdAt,
    });

    // 4. Emit 'intake_context_ready' for UI state updates
    this.chatNamespace.to(socketRoom).emit('intake_context_ready', {
      conversationId,
      uploadId,
      success: true,
      context: {
        vendorName: result.context.vendorName,
        solutionName: result.context.solutionName,
        solutionType: result.context.solutionType,
        industry: result.context.industry,
        features: result.context.features,
        claims: result.context.claims,
        complianceMentions: result.context.complianceMentions,
      },
      suggestedQuestions: result.suggestedQuestions,
      coveredCategories: result.coveredCategories,
      gapCategories: result.gapCategories,
      confidence: result.confidence,
    });
  } else {
    this.chatNamespace.to(socketRoom).emit('intake_context_ready', {
      conversationId,
      uploadId,
      success: false,
      context: null,
      error: result.error || 'Failed to extract context from document',
    });
  }
}

/**
 * Format intake context as assistant message for conversation history
 *
 * @param context - IntakeContext with vendor/solution info
 * @param gapCategories - From IntakeParseResult (not on IntakeContext per Story 1.2)
 */
private formatIntakeContextMessage(
  context: IntakeContext,
  gapCategories: string[]
): string {
  const parts: string[] = [
    '📄 **Document Context Extracted**',
    '',
  ];

  if (context.vendorName) {
    parts.push(`**Vendor:** ${context.vendorName}`);
  }
  if (context.solutionName) {
    parts.push(`**Solution:** ${context.solutionName}`);
  }
  if (context.solutionType) {
    parts.push(`**Type:** ${context.solutionType}`);
  }

  if (context.features.length > 0) {
    parts.push('', '**Key Features:**');
    context.features.slice(0, 5).forEach(f => parts.push(`- ${f}`));
    if (context.features.length > 5) {
      parts.push(`- ... and ${context.features.length - 5} more`);
    }
  }

  if (context.complianceMentions.length > 0) {
    parts.push('', `**Compliance Mentions:** ${context.complianceMentions.join(', ')}`);
  }

  // gapCategories comes from IntakeParseResult, not IntakeContext
  if (gapCategories && gapCategories.length > 0) {
    parts.push('', `**Areas Needing Clarification:** ${gapCategories.join(', ')}`);
  }

  return parts.join('\n');
}
```

```typescript
/**
 * IntakeContextCard - Shows extracted context from uploaded document
 *
 * Part of Epic 16: Document Parser Infrastructure
 */

'use client';

import React from 'react';
import { FileText, Building2, Package, CheckCircle, AlertCircle } from 'lucide-react';

interface IntakeContextSummary {
  vendorName: string | null;
  solutionName: string | null;
  solutionType: string | null;
  featureCount: number;
  coveredCategories: string[];
  gapCategories: string[];
}

interface IntakeContextCardProps {
  summary: IntakeContextSummary;
  confidence: number;
}

export function IntakeContextCard({ summary, confidence }: IntakeContextCardProps) {
  return (
    <div className="bg-blue-50 border border-blue-200 rounded-lg p-4 my-4">
      <div className="flex items-start gap-3">
        <FileText className="h-5 w-5 text-blue-600 mt-0.5" />
        <div className="flex-1">
          <h4 className="font-medium text-blue-900">Document Context Extracted</h4>

          <div className="mt-2 space-y-1 text-sm text-blue-800">
            {summary.vendorName && (
              <div className="flex items-center gap-2">
                <Building2 className="h-4 w-4" />
                <span>{summary.vendorName}</span>
              </div>
            )}
            {summary.solutionName && (
              <div className="flex items-center gap-2">
                <Package className="h-4 w-4" />
                <span>{summary.solutionName}</span>
                {summary.solutionType && (
                  <span className="text-blue-600">({summary.solutionType})</span>
                )}
              </div>
            )}
          </div>

          {/* Coverage indicators */}
          <div className="mt-3 flex flex-wrap gap-2">
            {summary.coveredCategories.slice(0, 3).map((cat) => (
              <span
                key={cat}
                className="inline-flex items-center gap-1 px-2 py-0.5 bg-green-100 text-green-700 text-xs rounded"
              >
                <CheckCircle className="h-3 w-3" />
                {cat.replace('_', ' ')}
              </span>
            ))}
            {summary.gapCategories.slice(0, 3).map((cat) => (
              <span
                key={cat}
                className="inline-flex items-center gap-1 px-2 py-0.5 bg-yellow-100 text-yellow-700 text-xs rounded"
              >
                <AlertCircle className="h-3 w-3" />
                {cat.replace('_', ' ')}
              </span>
            ))}
          </div>

          {/* Confidence */}
          <div className="mt-2 text-xs text-blue-600">
            Extraction confidence: {Math.round(confidence * 100)}%
          </div>
        </div>
      </div>
    </div>
  );
}
```
