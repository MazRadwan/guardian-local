**Sprint:** 5 - Testing & Integration
**Status:** pending

## Objective

Create integration tests for the document upload pipeline, testing **HTTP multipart upload** with **WebSocket progress/completion events**.

---

## Prerequisites

- Complete Story 5.1
- Review `DocumentUploadController` (Story 2.1)
- Review locked architecture: HTTP POST `/api/documents/upload` + WS `/chat` events

---

## Key Architecture Notes

- **Test runner:** Jest (NOT Vitest)
- **Upload transport:** HTTP multipart `POST /api/documents/upload` (NOT WebSocket binary)
- **Progress events:** WS `/chat` namespace emits `upload_progress`, `intake_context_ready`, `scoring_parse_ready`
- **Correlation:** Events filtered by `uploadId` returned in 202 response
- **Auth:** Bearer token in HTTP header + WS handshake
- **Deterministic:** Stub `IIntakeDocumentParser` / `IScoringDocumentParser` (no Anthropic calls)

// packages/backend/__tests__/integration/DocumentUploadPipeline.test.ts

import express from 'express';
import { createServer, type Server as HTTPServer } from 'http';
import request from 'supertest';
import { Server as SocketIOServer } from 'socket.io';
import { io as ioClient, type Socket as ClientSocket } from 'socket.io-client';
import jwt from 'jsonwebtoken';

import { createDocumentRoutes } from '../../src/infrastructure/http/routes/document.routes.js';
import { DocumentUploadController } from '../../src/infrastructure/http/controllers/DocumentUploadController.js';
import { AuthService } from '../../src/application/services/AuthService.js';
import { JWTProvider } from '../../src/infrastructure/auth/JWTProvider.js';
import { User } from '../../src/domain/entities/User.js';

describe('Document upload pipeline (HTTP + WS integration)', () => {
  let app: express.Express;
  let httpServer: HTTPServer;
  let ioServer: SocketIOServer;
  let client: ClientSocket;

  const JWT_SECRET = 'test-jwt-secret';
  const user = User.fromPersistence({
    id: 'user-123',
    email: 'test@guardian.com',
    passwordHash: 'hash',
    name: 'Test User',
    role: 'analyst',
    lastLoginAt: null,
    createdAt: new Date(),
    updatedAt: new Date(),
  });

  const userRepo = {
    findByEmail: jest.fn(),
    findById: jest.fn(async (id: string) => (id === user.id ? user : null)),
    create: jest.fn(async (u: any) => u),
    update: jest.fn(async (u: any) => u),
    delete: jest.fn(),
    findByUserId: jest.fn(),
  };

  const tokenProvider = new JWTProvider(JWT_SECRET, '1h');
  const authService = new AuthService(userRepo as any, tokenProvider);
  const token = tokenProvider.generateToken({ userId: user.id, email: user.getEmail(), role: user.role });

  // Mocks (fast + deterministic - no Anthropic calls)
  const mockFileStorage = {
    store: jest.fn(async () => '/internal/storage/path/file.pdf'),
    retrieve: jest.fn(),
    delete: jest.fn(),
    exists: jest.fn(),
  };

  const mockFileValidator = {
    validate: jest.fn(async () => ({ valid: true, documentType: 'pdf', warnings: [] })),
  };

  const mockConversationRepo = {
    findById: jest.fn(async () => ({ id: 'conv-123', userId: user.id })),
  };

  const mockConversationService = {
    sendMessage: jest.fn(async () => ({
      id: 'msg-1',
      conversationId: 'conv-123',
      role: 'assistant',
      content: { text: '📄 **Document Context Extracted**\n\n**Vendor:** Acme' },
      createdAt: new Date(),
    })),
  };

  const mockIntakeParser = {
    parseForContext: jest.fn(async () => ({
      success: true,
      confidence: 0.9,
      context: {
        vendorName: 'Acme',
        solutionName: null,
        solutionType: null,
        industry: null,
        features: [],
        claims: [],
        integrations: [],
        complianceMentions: [],
        architectureNotes: [],
        securityMentions: [],
        rawTextExcerpt: '',
        sourceFilePath: '',
      },
      suggestedQuestions: [],
      coveredCategories: [],
      gapCategories: ['privacy_risk'],
    })),
  };

  const mockScoringParser = {
    parseForResponses: jest.fn(async () => ({
      success: true,
      confidence: 0.8,
      assessmentId: 'assess-123',
      vendorName: 'Acme',
      solutionName: null,
      responses: [],
      expectedQuestionCount: null,
      parsedQuestionCount: 0,
      unparsedQuestions: [],
      isComplete: false,
    })),
  };

  beforeAll(async () => {
    app = express();

    httpServer = createServer(app);
    ioServer = new SocketIOServer(httpServer, {
      cors: { origin: '*', methods: ['GET', 'POST'] },
    });

    // Minimal /chat namespace auth + room join (matches Epic 16 contract)
    const chatNamespace = ioServer.of('/chat');
    chatNamespace.use((socket, next) => {
      const t = socket.handshake.auth?.token;
      if (!t) return next(new Error('Authentication token required'));
      try {
        const decoded = jwt.verify(t, JWT_SECRET) as any;
        (socket as any).userId = decoded.userId;
        socket.join(`user:${decoded.userId}`);
        next();
      } catch {
        next(new Error('Invalid authentication token'));
      }
    });

    const controller = new DocumentUploadController(
      mockFileStorage as any,
      mockFileValidator as any,
      mockIntakeParser as any,
      mockScoringParser as any,
      mockConversationRepo as any,
      mockConversationService as any,
      chatNamespace
    );

    app.use('/api/documents', createDocumentRoutes(controller, authService));

    await new Promise<void>((resolve) => httpServer.listen(0, resolve));
    const port = (httpServer.address() as any).port;

    client = ioClient(`http://localhost:${port}/chat`, {
      auth: { token },
      transports: ['websocket'],
    });

    await new Promise<void>((resolve, reject) => {
      client.on('connect', () => resolve());
      client.on('connect_error', reject);
    });
  });

  afterAll(async () => {
    client.disconnect();
    ioServer.close();
    httpServer.close();
  });

  beforeEach(() => {
    jest.clearAllMocks();
  });

  describe('intake mode', () => {
    it('POST returns 202 and WS emits progress + message + intake_context_ready (by uploadId)', async () => {
      const progress: any[] = [];
      const receivedMessages: any[] = [];
      let intakeReady: any | null = null;

      client.on('upload_progress', (e) => progress.push(e));
      client.on('message', (m) => receivedMessages.push(m));
      client.on('intake_context_ready', (e) => (intakeReady = e));

      const res = await request(httpServer)
        .post('/api/documents/upload')
        .set('Authorization', `Bearer ${token}`)
        .field('conversationId', 'conv-123')
        .field('mode', 'intake')
        .attach('file', Buffer.from('%PDF-1.4\n%Mock'), {
          filename: 'test.pdf',
          contentType: 'application/pdf',
        });

      expect(res.status).toBe(202);
      expect(res.body.uploadId).toBeDefined();

      // Wait until intake_context_ready arrives for that uploadId
      await new Promise<void>((resolve, reject) => {
        const timeout = setTimeout(() => reject(new Error('Timeout waiting for intake_context_ready')), 3000);
        const check = () => {
          if (intakeReady?.uploadId === res.body.uploadId) {
            clearTimeout(timeout);
            resolve();
          }
        };
        const interval = setInterval(() => {
          check();
          if (intakeReady) clearInterval(interval);
        }, 25);
      });

      expect(intakeReady).toMatchObject({
        conversationId: 'conv-123',
        uploadId: res.body.uploadId,
        success: true,
      });

      // Progress correlation by uploadId
      expect(progress.some((p) => p.uploadId === res.body.uploadId && p.stage === 'storing')).toBe(true);
      expect(progress.some((p) => p.uploadId === res.body.uploadId && p.stage === 'parsing')).toBe(true);
      expect(progress.some((p) => p.uploadId === res.body.uploadId && p.stage === 'complete')).toBe(true);

      // Context message emitted to chat UI
      expect(receivedMessages.some((m) => m.role === 'assistant')).toBe(true);
    });
  });

  describe('scoring mode', () => {
    it('emits scoring_parse_ready (by uploadId)', async () => {
      let scoringReady: any | null = null;
      client.on('scoring_parse_ready', (e) => (scoringReady = e));

      const res = await request(httpServer)
        .post('/api/documents/upload')
        .set('Authorization', `Bearer ${token}`)
        .field('conversationId', 'conv-123')
        .field('mode', 'scoring')
        .attach('file', Buffer.from('%PDF-1.4\n%Mock'), {
          filename: 'questionnaire.pdf',
          contentType: 'application/pdf',
        });

      expect(res.status).toBe(202);

      await new Promise<void>((resolve, reject) => {
        const timeout = setTimeout(() => reject(new Error('Timeout waiting for scoring_parse_ready')), 3000);
        const interval = setInterval(() => {
          if (scoringReady?.uploadId === res.body.uploadId) {
            clearTimeout(timeout);
            clearInterval(interval);
            resolve();
          }
        }, 25);
      });

      expect(scoringReady).toMatchObject({
        success: true,
        uploadId: res.body.uploadId,
        assessmentId: 'assess-123',
      });
    });
  });

  describe('authentication', () => {
    it('rejects upload without Authorization header', async () => {
      const res = await request(httpServer)
        .post('/api/documents/upload')
        .field('conversationId', 'conv-123')
        .field('mode', 'intake')
        .attach('file', Buffer.from('%PDF-1.4'), {
          filename: 'test.pdf',
          contentType: 'application/pdf',
        });

      expect(res.status).toBe(401);
    });

    it('rejects upload with invalid token', async () => {
      const res = await request(httpServer)
        .post('/api/documents/upload')
        .set('Authorization', 'Bearer invalid-token')
        .field('conversationId', 'conv-123')
        .field('mode', 'intake')
        .attach('file', Buffer.from('%PDF-1.4'), {
          filename: 'test.pdf',
          contentType: 'application/pdf',
        });

      expect(res.status).toBe(401);
    });
  });

  describe('validation', () => {
    it('rejects file when validator returns invalid (exercises FileValidationService layer)', async () => {
      // NOTE: We send an ALLOWED MIME type (application/pdf) so Multer's fileFilter passes,
      // but mock the validator to return invalid. This exercises the FileValidationService layer.
      // Sending application/x-executable would be rejected by Multer before reaching the validator.
      mockFileValidator.validate.mockResolvedValueOnce({
        valid: false,
        documentType: null,
        error: 'File appears corrupt or invalid',
        warnings: [],
      });

      const res = await request(httpServer)
        .post('/api/documents/upload')
        .set('Authorization', `Bearer ${token}`)
        .field('conversationId', 'conv-123')
        .field('mode', 'intake')
        .attach('file', Buffer.from('not-a-real-pdf'), {
          filename: 'fake.pdf',
          contentType: 'application/pdf',  // Allowed MIME so Multer passes
        });

      expect(res.status).toBe(400);
      expect(res.body.error).toContain('corrupt');
    });

    it('rejects missing conversationId', async () => {
      const res = await request(httpServer)
        .post('/api/documents/upload')
        .set('Authorization', `Bearer ${token}`)
        .field('mode', 'intake')
        .attach('file', Buffer.from('%PDF-1.4'), {
          filename: 'test.pdf',
          contentType: 'application/pdf',
        });

      expect(res.status).toBe(400);
    });
  });
});
