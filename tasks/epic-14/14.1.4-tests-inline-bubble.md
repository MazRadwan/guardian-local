### 2. State Transition Tests

Add integration tests:

```tsx
describe('Questionnaire State Transitions (Story 14.1)', () => {
  beforeEach(() => {
    useChatStore.setState({
      messages: [],
      pendingQuestionnaire: null,
      questionnaireUIState: 'hidden',
      questionnaireMessageIndex: -1,
      isGeneratingQuestionnaire: false,
    });
  });

  it('full flow: hidden -> ready -> generating -> download', () => {
    const { result } = renderHook(() => useChatStore());

    // Add messages first
    act(() => {
      result.current.addMessage({ role: 'user', content: 'Assess vendor X' });
      result.current.addMessage({ role: 'assistant', content: 'Collecting info...' });
    });

    // Receive questionnaire_ready event (simulated)
    act(() => {
      result.current.setPendingQuestionnaire({
        conversationId: 'conv-123',
        assessmentType: 'comprehensive',
        vendorName: 'Vendor X',
        solutionName: null,
        contextSummary: null,
        estimatedQuestions: 85,
        selectedCategories: null,
      });
      result.current.setQuestionnaireUIState('ready');
    });

    expect(result.current.questionnaireUIState).toBe('ready');
    expect(result.current.questionnaireMessageIndex).toBe(2); // After 2 messages

    // User clicks Generate
    act(() => {
      result.current.setQuestionnaireUIState('generating');
      result.current.setGenerating(true);
    });

    expect(result.current.questionnaireUIState).toBe('generating');
    expect(result.current.isGeneratingQuestionnaire).toBe(true);

    // Generation complete - export_ready event
    act(() => {
      result.current.setExportReady('conv-123', {
        conversationId: 'conv-123',
        assessmentId: 'assessment-123',
        formats: ['pdf', 'word', 'excel'],
        questionCount: 85,
      });
      result.current.setQuestionnaireUIState('download');
      result.current.setGenerating(false);
    });

    expect(result.current.questionnaireUIState).toBe('download');
    expect(result.current.isGeneratingQuestionnaire).toBe(false);
    expect(result.current.questionnaireMessageIndex).toBe(2); // Still at same position
  });

  it('position persists across state transitions', () => {
    const { result } = renderHook(() => useChatStore());

    // Setup initial state
    act(() => {
      result.current.addMessage({ role: 'user', content: 'Test 1' });
      result.current.addMessage({ role: 'assistant', content: 'Test 2' });
      result.current.setPendingQuestionnaire({
        conversationId: 'conv-123',
        assessmentType: 'comprehensive',
        vendorName: 'Test',
        solutionName: null,
        contextSummary: null,
        estimatedQuestions: 85,
        selectedCategories: null,
      });
    });

    const initialPosition = result.current.questionnaireMessageIndex;

    // Add more messages (simulating conversation continuing)
    act(() => {
      result.current.addMessage({ role: 'user', content: 'Test 3' });
      result.current.addMessage({ role: 'assistant', content: 'Test 4' });
    });

    // Position should NOT change
    expect(result.current.questionnaireMessageIndex).toBe(initialPosition);
  });

  it('conversation switch clears questionnaire state', () => {
    const { result } = renderHook(() => useChatStore());

    // Setup questionnaire
    act(() => {
      result.current.setPendingQuestionnaire({
        conversationId: 'conv-123',
        assessmentType: 'comprehensive',
        vendorName: 'Test',
        solutionName: null,
        contextSummary: null,
        estimatedQuestions: 85,
        selectedCategories: null,
      });
      result.current.setQuestionnaireUIState('ready');
    });

    expect(result.current.questionnaireMessageIndex).toBeGreaterThanOrEqual(0);

    // Switch conversation
    act(() => {
      result.current.setActiveConversation('conv-456');
    });

    // Should clear questionnaire state
    expect(result.current.pendingQuestionnaire).toBeNull();
    expect(result.current.questionnaireMessageIndex).toBe(-1);
  });
});
```

```tsx
describe('Rehydration with Inline Positioning (Story 14.1)', () => {
  it('rehydrated questionnaire appears at stored position', () => {
    // This test verifies that when a questionnaire is rehydrated from localStorage,
    // it appears at the correct position in the message list.
    //
    // Note: The position is captured when setPendingQuestionnaire is called,
    // but on rehydration we may need to recalculate based on message count.

    const { result: persistenceResult } = renderHook(() =>
      useQuestionnairePersistence('user-123')
    );

    // Save a payload
    act(() => {
      persistenceResult.current.savePayload('conv-123', {
        conversationId: 'conv-123',
        assessmentType: 'comprehensive',
        vendorName: 'Test',
        solutionName: null,
        contextSummary: null,
        estimatedQuestions: 85,
        selectedCategories: null,
      });
    });

    // Load it back
    const loaded = persistenceResult.current.loadPayload('conv-123');

    expect(loaded).not.toBeNull();
    expect(loaded?.conversationId).toBe('conv-123');
  });

  it('no duplicate questionnaire entries on rapid conversation switches', () => {
    const { result } = renderHook(() => useChatStore());

    // Setup questionnaire in conv-123
    act(() => {
      result.current.setActiveConversation('conv-123');
      result.current.setPendingQuestionnaire({
        conversationId: 'conv-123',
        assessmentType: 'comprehensive',
        vendorName: 'Test',
        solutionName: null,
        contextSummary: null,
        estimatedQuestions: 85,
        selectedCategories: null,
      });
      result.current.setQuestionnaireUIState('ready');
    });

    // Rapid conversation switches
    act(() => {
      result.current.setActiveConversation('conv-456');
    });
    act(() => {
      result.current.setActiveConversation('conv-123');
    });
    act(() => {
      result.current.setActiveConversation('conv-456');
    });
    act(() => {
      result.current.setActiveConversation('conv-123');
    });

    // Should only have one pendingQuestionnaire (or none if cleared)
    // The key is no duplicates in render
    expect(result.current.questionnaireMessageIndex).toBeDefined();
  });
});
```
