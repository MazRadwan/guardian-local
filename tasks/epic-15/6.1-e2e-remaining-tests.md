## Mock Patterns for E2E Tests

> **IMPORTANT:** E2E tests should NOT call real LLM APIs.
> Mock the ScoringService or LLM client to return deterministic results.
> This prevents flaky tests and reduces CI runtime.

### MockLLMClient

**File:** `packages/backend/src/__tests__/mocks/MockLLMClient.ts`

```typescript
import { ILLMClient, StreamWithToolOptions } from '../../application/interfaces/ILLMClient';
import { ALL_DIMENSIONS } from '../../domain/scoring/rubric';

/**
 * Mock LLM client for E2E tests - returns deterministic scoring results
 */
export class MockLLMClient implements ILLMClient {
  private modelId = 'mock-claude-test';

  async streamWithTool(options: StreamWithToolOptions): Promise<void> {
    // Simulate streaming narrative
    const narrative = `## Executive Summary\n\nThis is a mock assessment for testing purposes.\n\n## Risk Assessment\n\nAll dimensions evaluated with deterministic scores.`;

    // Stream text deltas
    for (const chunk of narrative.split(' ')) {
      options.onTextDelta?.(chunk + ' ');
      await new Promise(resolve => setTimeout(resolve, 10)); // Simulate delay
    }

    // Call scoring_complete tool with deterministic payload
    const mockPayload = {
      compositeScore: 72,
      recommendation: 'conditional',
      overallRiskRating: 'medium',
      executiveSummary: 'Mock assessment for testing.',
      keyFindings: ['Finding 1', 'Finding 2'],
      disqualifyingFactors: [],
      dimensionScores: ALL_DIMENSIONS.map((d, i) => ({
        dimension: d,
        score: 60 + (i * 4), // Deterministic scores: 60, 64, 68, ...
        riskRating: 'medium' as const,
      })),
    };

    options.onToolUse?.('scoring_complete', mockPayload);
  }

  getModelId(): string {
    return this.modelId;
  }
}
```

```typescript
// In test setup, inject mock instead of real client
```

## E2E Test: Full Scoring Flow

> **NOTE:** Uses MockLLMClient for deterministic, fast tests.
> Timeout reduced from 120s to 30s since no real LLM calls.

**File:** `packages/backend/src/__tests__/e2e/scoring-flow.e2e.test.ts`

```typescript
import { createTestApp } from '../helpers/testApp';
import { io as Client, Socket } from 'socket.io-client';
import { createTestUser, createTestAssessment, createTestQuestionnaire } from '../helpers/fixtures';
import { MockLLMClient } from '../mocks/MockLLMClient';
import * as fs from 'fs/promises';
import * as path from 'path';

describe('E2E: Scoring Flow', () => {
  let app: TestApp;
  let clientSocket: Socket;
  let testUser: TestUser;
  let testAssessment: TestAssessment;

  beforeAll(async () => {
    // Use MockLLMClient for deterministic tests
    app = await createTestApp({
      llmClient: new MockLLMClient(),
    });
    testUser = await createTestUser(app.db);
    testAssessment = await createTestAssessment(app.db, testUser.id);

    // Generate and export a test questionnaire
    await createTestQuestionnaire(app.db, testAssessment.id);
  });

  beforeEach((done) => {
    clientSocket = Client(app.wsUrl, {
      auth: { token: testUser.token },
    });
    clientSocket.on('connect', done);
  });

  afterEach(() => {
    clientSocket.close();
  });

  afterAll(async () => {
    await app.cleanup();
  });

  describe('Complete scoring workflow', () => {
    it('should score a completed questionnaire end-to-end', async () => {
      // 1. Switch to scoring mode
      const modeChangedPromise = new Promise<void>((resolve) => {
        clientSocket.on('mode_changed', (data) => {
          expect(data.mode).toBe('scoring');
          resolve();
        });
      });

      clientSocket.emit('mode_change', { mode: 'scoring' });
      await modeChangedPromise;

      // 2. Upload completed questionnaire file
      const testFilePath = path.join(__dirname, '../fixtures/completed-questionnaire.pdf');
      const fileBuffer = await fs.readFile(testFilePath);

      // Upload via HTTP
      const uploadResponse = await app.request
        .post('/api/files/upload')
        .set('Authorization', `Bearer ${testUser.token}`)
        .attach('file', fileBuffer, 'questionnaire.pdf')
        .field('assessmentId', testAssessment.id)
        .field('mode', 'scoring');

      expect(uploadResponse.status).toBe(200);
      const { fileId } = uploadResponse.body;

      // 3. Trigger scoring and track progress
      const progressEvents: string[] = [];
      const scoringCompletePromise = new Promise<ScoringResultData>((resolve, reject) => {
        clientSocket.on('scoring_progress', (event) => {
          progressEvents.push(event.status);
        });

        clientSocket.on('scoring_complete', (result) => {
          resolve(result);
        });

        clientSocket.on('scoring_error', (error) => {
          reject(new Error(error.error));
        });

        // With mock client, timeout can be shorter
        setTimeout(() => reject(new Error('Scoring timeout')), 30000);
      });

      clientSocket.emit('scoring_upload', {
        fileId,
        assessmentId: testAssessment.id,
      });

      const result = await scoringCompletePromise;

      // 4. Verify progress events
      expect(progressEvents).toContain('parsing');
      expect(progressEvents).toContain('scoring');
      expect(progressEvents).toContain('validating');
      expect(progressEvents).toContain('complete');

      // 5. Verify scoring results (deterministic from mock)
      expect(result.compositeScore).toBe(72); // Mock returns 72
      expect(result.recommendation).toBe('conditional');
      expect(result.overallRiskRating).toBe('medium');
      expect(result.dimensionScores).toHaveLength(10);
      expect(result.executiveSummary).toBeTruthy();
      expect(result.batchId).toBeTruthy();

      // 6. Verify dimension scores
      for (const dimension of result.dimensionScores) {
        expect(dimension.score).toBeGreaterThanOrEqual(0);
        expect(dimension.score).toBeLessThanOrEqual(100);
        expect(dimension.riskRating).toBe('medium'); // Mock always returns medium
      }

      // 7. Verify data persisted to database
      const savedResult = await app.db.query.assessmentResults.findFirst({
        where: (r, { eq }) => eq(r.batchId, result.batchId),
      });
      expect(savedResult).toBeTruthy();
      expect(savedResult!.compositeScore).toBe(result.compositeScore);

      const savedScores = await app.db.query.dimensionScores.findMany({
        where: (s, { eq }) => eq(s.batchId, result.batchId),
      });
      expect(savedScores).toHaveLength(10);

      // 8. Verify export works
      const pdfResponse = await app.request
        .get(`/api/export/scoring/${result.batchId}/pdf`)
        .set('Authorization', `Bearer ${testUser.token}`);

      expect(pdfResponse.status).toBe(200);
      expect(pdfResponse.headers['content-type']).toBe('application/pdf');
    }, 60000); // 60s timeout with mock (down from 150s)

    it('should reject questionnaire without assessment ID', async () => {
      // Switch to scoring mode
      clientSocket.emit('mode_change', { mode: 'scoring' });
      await new Promise((resolve) => clientSocket.on('mode_changed', resolve));

      // Upload a file without assessment ID
      const testFilePath = path.join(__dirname, '../fixtures/no-assessment-id.pdf');
      const fileBuffer = await fs.readFile(testFilePath);

      const uploadResponse = await app.request
        .post('/api/files/upload')
        .set('Authorization', `Bearer ${testUser.token}`)
        .attach('file', fileBuffer, 'questionnaire.pdf')
        .field('mode', 'scoring');

      const { fileId } = uploadResponse.body;

      const errorPromise = new Promise<string>((resolve) => {
        clientSocket.on('scoring_error', (error) => {
          resolve(error.error);
        });
      });

      clientSocket.emit('scoring_upload', {
        fileId,
        assessmentId: 'unknown-assessment',
      });

      const errorMessage = await errorPromise;
      expect(errorMessage).toContain('Assessment ID');
    });

    it('should handle abort during scoring', async () => {
      clientSocket.emit('mode_change', { mode: 'scoring' });
      await new Promise((resolve) => clientSocket.on('mode_changed', resolve));

      // Start scoring
      const testFilePath = path.join(__dirname, '../fixtures/completed-questionnaire.pdf');
      const fileBuffer = await fs.readFile(testFilePath);

      const uploadResponse = await app.request
        .post('/api/files/upload')
        .set('Authorization', `Bearer ${testUser.token}`)
        .attach('file', fileBuffer, 'questionnaire.pdf')
        .field('assessmentId', testAssessment.id)
        .field('mode', 'scoring');

      const { fileId } = uploadResponse.body;

      const resultPromise = new Promise<{ aborted: boolean }>((resolve) => {
        clientSocket.on('scoring_progress', (event) => {
          if (event.status === 'scoring') {
            // Abort when scoring starts
            clientSocket.emit('scoring_abort', { conversationId: clientSocket.id });
          }
          if (event.error?.includes('aborted')) {
            resolve({ aborted: true });
          }
        });

        clientSocket.on('scoring_error', (error) => {
          if (error.error.includes('aborted')) {
            resolve({ aborted: true });
          }
        });
      });

      clientSocket.emit('scoring_upload', { fileId, assessmentId: testAssessment.id });

      const result = await resultPromise;
      expect(result.aborted).toBe(true);
    });
  });
});
```

```typescript
    expect(container.firstChild).toBeNull();

    expect(screen.getByText('Analyzing...')).toBeInTheDocument();
  });

  it('shows loading spinner when not complete', () => {
    render(<RotatingStatus status="scoring" currentMessage="Processing" />);
    // Look for the spinner element
    expect(document.querySelector('.animate-spin')).toBeInTheDocument();
  });

  it('does not show spinner when complete', () => {
    render(<RotatingStatus status="complete" currentMessage="Done!" />);
    expect(document.querySelector('.animate-spin')).not.toBeInTheDocument();
  });

  it('rotates through messages over time', () => {
    render(<RotatingStatus status="scoring" currentMessage="" />);

    // Initial message
    expect(screen.getByText(/Analyzing|Evaluating|may take/)).toBeInTheDocument();

    // Advance timer
    act(() => {
      jest.advanceTimersByTime(8000);
    });

    // Should show a different message or cycle
    expect(screen.getByText(/Analyzing|Evaluating|may take|Almost/)).toBeInTheDocument();
  });

  it('resets message index when status changes', () => {
    const { rerender } = render(<RotatingStatus status="parsing" currentMessage="" />);

    act(() => {
      jest.advanceTimersByTime(16000); // Multiple rotations
    });

    // Change status
    rerender(<RotatingStatus status="scoring" currentMessage="" />);
```
