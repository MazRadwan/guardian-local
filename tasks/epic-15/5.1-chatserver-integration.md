## WebSocket Events

### Client → Server Events

| Event | Payload | Description |
|-------|---------|-------------|
| `mode_change` | `{ mode: ConversationMode }` | User changes mode |
| `scoring_upload` | `{ fileId: string, assessmentId: string }` | Upload completed questionnaire |
| `scoring_abort` | `{ conversationId: string }` | Cancel in-progress scoring |
| `scoring_export` | `{ batchId: string, format: 'pdf' \| 'word' }` | Export scoring report |

### Server → Client Events

| Event | Payload | Description |
|-------|---------|-------------|
| `mode_changed` | `{ mode: ConversationMode }` | Mode change confirmed |
| `scoring_welcome` | `{ message: string }` | Welcome message for scoring mode |
| `scoring_progress` | `ScoringProgressEvent` | Progress updates |
| `scoring_complete` | `ScoringResultData` | Scoring results |
| `scoring_error` | `{ error: string }` | Scoring failed |
| `scoring_export_ready` | `{ url: string, format: string }` | Export download ready |

## ChatServer Modifications

**File:** `packages/backend/src/infrastructure/websocket/ChatServer.ts`

Add scoring mode handling:

```typescript
import { IScoringService } from '../../application/interfaces/IScoringService';
import { IScoringExportService } from '../../application/interfaces/IScoringExportService';
import { ScoringProgressEvent, ScoringResultData } from '../../domain/scoring/types';
import { getSystemPrompt } from '../ai/prompts';

export class ChatServer {
  private scoringService: IScoringService;           // Use interface, not concrete
  private scoringExportService: IScoringExportService; // Use interface

  constructor(
    // ... existing dependencies
    scoringService: IScoringService,
    scoringExportService: IScoringExportService,
  ) {
    // ... existing initialization
    this.scoringService = scoringService;
    this.scoringExportService = scoringExportService;
  }

  private setupEventHandlers(socket: Socket): void {
    // ... existing handlers

    // Mode change handler
    socket.on('mode_change', async (data: { mode: ConversationMode }) => {
      await this.handleModeChange(socket, data.mode);
    });

    // Scoring upload handler
    socket.on('scoring_upload', async (data: { fileId: string; assessmentId: string }) => {
      await this.handleScoringUpload(socket, data);
    });

    // Scoring abort handler
    socket.on('scoring_abort', (data: { conversationId: string }) => {
      this.scoringService.abort(data.conversationId);
    });

    // Scoring export handler
    socket.on('scoring_export', async (data: { batchId: string; format: 'pdf' | 'word' }) => {
      await this.handleScoringExport(socket, data);
    });
  }

  /**
   * Handle mode change - emit welcome message for scoring mode
   */
  private async handleModeChange(socket: Socket, mode: ConversationMode): Promise<void> {
    const conversationId = socket.data.conversationId;

    // Update conversation mode
    await this.conversationService.updateMode(conversationId, mode);

    // Emit mode changed confirmation
    socket.emit('mode_changed', { mode });

    // Send welcome message for scoring mode
    if (mode === 'scoring') {
      const welcomeMessage = this.getScoringWelcomeMessage();
      socket.emit('scoring_welcome', { message: welcomeMessage });

      // Also add to conversation as assistant message
      await this.messageService.create({
        conversationId,
        role: 'assistant',
        content: welcomeMessage,
      });
    }
  }

  /**
   * Handle scoring upload - trigger scoring workflow
   */
  private async handleScoringUpload(
    socket: Socket,
    data: { fileId: string; assessmentId: string }
  ): Promise<void> {
    const conversationId = socket.data.conversationId;
    const userId = socket.data.userId;

    try {
      // Execute scoring workflow with progress updates
      const result = await this.scoringService.score(
        {
          assessmentId: data.assessmentId,
          conversationId,
          fileId: data.fileId,
          userId,
        },
        (event: ScoringProgressEvent) => {
          socket.emit('scoring_progress', event);
        }
      );

      if (result.success && result.report) {
        // Emit scoring complete with results
        const resultData: ScoringResultData = {
          compositeScore: result.report.payload.compositeScore,
          recommendation: result.report.payload.recommendation,
          overallRiskRating: result.report.payload.overallRiskRating,
          executiveSummary: result.report.payload.executiveSummary,
          keyFindings: result.report.payload.keyFindings,
          dimensionScores: result.report.payload.dimensionScores,
          batchId: result.batchId,
          assessmentId: data.assessmentId,
        };

        socket.emit('scoring_complete', resultData);

        // Store narrative report as assistant message
        await this.messageService.create({
          conversationId,
          role: 'assistant',
          content: result.report.narrativeReport,
          metadata: {
            type: 'scoring_report',
            batchId: result.batchId,
            compositeScore: result.report.payload.compositeScore,
          },
        });
      } else {
        socket.emit('scoring_error', { error: result.error || 'Scoring failed' });
      }
    } catch (error) {
      const errorMessage = error instanceof Error ? error.message : 'Unknown error';
      socket.emit('scoring_error', { error: errorMessage });
    }
  }

  /**
   * Handle scoring export request
   *
   * NOTE: IScoringExportService.generateExport() returns a URL or buffer.
   * Ensure the interface matches what ExportController uses.
   */
  private async handleScoringExport(
    socket: Socket,
    data: { batchId: string; format: 'pdf' | 'word' }
  ): Promise<void> {
    const userId = socket.data.userId;

    try {
      // Use consistent method names with ExportController
      const url = data.format === 'pdf'
        ? await this.scoringExportService.generatePDFUrl(data.batchId, userId)
        : await this.scoringExportService.generateWordUrl(data.batchId, userId);

      socket.emit('scoring_export_ready', {
        url,
        format: data.format,
      });
    } catch (error) {
      const errorMessage = error instanceof Error ? error.message : 'Export failed';
      socket.emit('scoring_error', { error: errorMessage });
    }
  }

  /**
   * Get welcome message for scoring mode
   */
  private getScoringWelcomeMessage(): string {
    return `## Scoring Mode

Welcome to Scoring Mode. I'll help you assess vendor questionnaire responses against the Guardian rubric.

**To get started:**
1. Upload a completed Guardian questionnaire (PDF recommended)
2. I'll extract and analyze all responses
3. You'll receive a comprehensive risk assessment with scores across 10 dimensions

**Requirements:**
- The document must be a questionnaire exported from Guardian
- It must contain a valid Assessment ID in the header

**Supported formats:** PDF (recommended), Word (.docx)

Ready when you are - upload your completed questionnaire to begin.`;
  }
}
```

import { Request, Response } from 'express';
import { ScoringExportService } from '../../../application/services/ScoringExportService';

export class ExportController {
  constructor(
    // ... existing dependencies
    private scoringExportService: ScoringExportService,
  ) {}

  // ... existing methods

  /**
   * Export scoring report as PDF
   */
  exportScoringPDF = async (req: Request, res: Response): Promise<void> => {
    try {
      const { batchId } = req.params;
      const userId = req.user!.id;

      const buffer = await this.scoringExportService.generatePDF(batchId, userId);

      res.setHeader('Content-Type', 'application/pdf');
      res.setHeader(
        'Content-Disposition',
        `attachment; filename="guardian-risk-assessment-${batchId}.pdf"`
      );
      res.send(buffer);
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Export failed';
      res.status(500).json({ error: message });
    }
  };

  /**
   * Export scoring report as Word
   */
  exportScoringWord = async (req: Request, res: Response): Promise<void> => {
    try {
      const { batchId } = req.params;
      const userId = req.user!.id;

      const buffer = await this.scoringExportService.generateWord(batchId, userId);

      res.setHeader(
        'Content-Type',
        'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
      );
      res.setHeader(
        'Content-Disposition',
        `attachment; filename="guardian-risk-assessment-${batchId}.docx"`
      );
      res.send(buffer);
    } catch (error) {
      const message = error instanceof Error ? error.message : 'Export failed';
      res.status(500).json({ error: message });
    }
  };
}

import { createTestServer } from '../../helpers/testServer';
import { io as Client, Socket } from 'socket.io-client';

describe('ChatServer - Scoring Integration', () => {
  let serverUrl: string;
  let clientSocket: Socket;

  beforeAll(async () => {
    serverUrl = await createTestServer();
  });

  beforeEach((done) => {
    clientSocket = Client(serverUrl, {
      auth: { token: 'test-token' },
    });
    clientSocket.on('connect', done);
  });

  afterEach(() => {
    clientSocket.close();
  });

  describe('mode_change to scoring', () => {
    it('should emit scoring_welcome when switching to scoring mode', (done) => {
      clientSocket.on('scoring_welcome', (data) => {
        expect(data.message).toContain('Scoring Mode');
        expect(data.message).toContain('upload');
        done();
      });

      clientSocket.emit('mode_change', { mode: 'scoring' });
    });
  });

  describe('scoring_upload', () => {
    it('should emit progress events during scoring', (done) => {
      const progressEvents: string[] = [];

      clientSocket.on('scoring_progress', (event) => {
        progressEvents.push(event.status);
        if (event.status === 'complete') {
          expect(progressEvents).toContain('parsing');
          expect(progressEvents).toContain('scoring');
          done();
        }
      });

      clientSocket.emit('scoring_upload', {
        fileId: 'test-file-id',
        assessmentId: 'test-assessment-id',
      });
    });

    it('should emit scoring_complete with results on success', (done) => {
      clientSocket.on('scoring_complete', (result) => {
        expect(result.compositeScore).toBeDefined();
        expect(result.recommendation).toBeDefined();
        expect(result.dimensionScores).toHaveLength(10);
        done();
      });

      clientSocket.emit('scoring_upload', {
        fileId: 'test-file-id',
        assessmentId: 'test-assessment-id',
      });
    });
  });

  describe('scoring_abort', () => {
    it('should abort in-progress scoring', (done) => {
      clientSocket.on('scoring_progress', (event) => {
        if (event.status === 'scoring') {
          // Abort mid-scoring
          clientSocket.emit('scoring_abort', { conversationId: 'test-conv' });
        }
        if (event.error?.includes('aborted')) {
          done();
        }
      });

      clientSocket.emit('scoring_upload', {
        fileId: 'test-file-id',
        assessmentId: 'test-assessment-id',
      });
    });
  });
});
