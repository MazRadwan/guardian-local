#### Test Setup (Concrete)

```typescript
import { describe, it, expect, beforeAll, afterAll, beforeEach, vi } from 'vitest';
import { QuestionnaireGenerationService } from '../../src/application/services/QuestionnaireGenerationService.js';
import { questionnaireToMarkdown } from '../../src/infrastructure/rendering/questionnaireToMarkdown.js';
import { DrizzleQuestionRepository } from '../../src/infrastructure/repositories/DrizzleQuestionRepository.js';
import { DrizzleAssessmentRepository } from '../../src/infrastructure/repositories/DrizzleAssessmentRepository.js';
import { fixtureQuestionnaireSchema } from '../fixtures/questionnaireSchema.js';

/**
 * Integration tests for the hybrid questionnaire generation flow
 *
 * These tests verify:
 * 1. Service generates valid schema
 * 2. Markdown is deterministically derived from schema
 * 3. Assessment is created with correct questions
 * 4. Chat content matches export content
 */
describe('QuestionnaireGenerationService Integration', () => {
  let service: QuestionnaireGenerationService;
  let mockClaudeClient: {
    sendMessage: ReturnType<typeof vi.fn>;
  };

  beforeAll(async () => {
    // Mock Claude to return known JSON via tool_use
    mockClaudeClient = {
      sendMessage: vi.fn(),
    };
  });

  beforeEach(() => {
    // Reset mock before each test
    mockClaudeClient.sendMessage.mockReset();
  });

  describe('end-to-end generation flow', () => {
    it('generates schema, persists questions, and renders markdown', async () => {
      const expectedSchema = fixtureQuestionnaireSchema({
        assessmentType: 'comprehensive',
        vendorName: 'Test Vendor',
        solutionName: 'Test Solution',
      });

      mockClaudeClient.sendMessage.mockResolvedValue({
        content: '',
        toolUse: [{ id: 'tool_1', name: 'output_questionnaire', input: expectedSchema }],
      });

      const result = await service.generate({
        conversationId: 'test-conv-1',
        userId: 'test-user-1',
        assessmentType: 'comprehensive',
        vendorName: 'Test Vendor',
        solutionName: 'Test Solution',
      });

      // Verify schema structure
      expect(result.schema.version).toBe('1.0');
      expect(result.schema.metadata.assessmentType).toBe('comprehensive');
      expect(result.schema.sections.length).toBeGreaterThan(0);

      // Verify assessment created
      expect(result.assessmentId).toBeDefined();

      // Verify markdown rendered
      expect(result.markdown).toContain('# Test Vendor Assessment Questionnaire');
      expect(result.markdown).toContain('## Privacy Risk');
    });

    it('chat content matches export content', async () => {
      const expectedSchema = fixtureQuestionnaireSchema({ assessmentType: 'quick' });

      mockClaudeClient.sendMessage.mockResolvedValue({
        content: '',
        toolUse: [{ id: 'tool_1', name: 'output_questionnaire', input: expectedSchema }],
      });

      const result = await service.generate({
        conversationId: 'test-conv-2',
        userId: 'test-user-1',
        assessmentType: 'quick',
      });

      // Extract question count from both sources
      const schemaQuestionCount = result.schema.sections.reduce(
        (sum, section) => sum + section.questions.length,
        0
      );

      const markdownQuestionMatches = result.markdown.match(/^\d+\./gm) || [];
      const markdownQuestionCount = markdownQuestionMatches.length;

      // They should match exactly
      expect(markdownQuestionCount).toBe(schemaQuestionCount);
      expect(schemaQuestionCount).toBe(result.schema.metadata.questionCount);
    });

    it('handles category_focused assessment type', async () => {
      const expectedSchema = fixtureQuestionnaireSchema({
        assessmentType: 'category_focused',
        focusedDimensions: ['privacy_risk'],
      });

      mockClaudeClient.sendMessage.mockResolvedValue({
        content: '',
        toolUse: [{ id: 'tool_1', name: 'output_questionnaire', input: expectedSchema }],
      });

      const result = await service.generate({
        conversationId: 'test-conv-3',
        userId: 'test-user-1',
        assessmentType: 'category_focused',
        selectedCategories: ['privacy_risk'],
      });

      expect(result.schema.metadata.assessmentType).toBe('category_focused');
    });
  });

  describe('schema validation', () => {
    it('validates all 10 risk dimensions are recognized', async () => {
      const expectedSchema = fixtureQuestionnaireSchema({ assessmentType: 'comprehensive' });

      mockClaudeClient.sendMessage.mockResolvedValue({
        content: '',
        toolUse: [{ id: 'tool_1', name: 'output_questionnaire', input: expectedSchema }],
      });

      const result = await service.generate({
        conversationId: 'test-conv-4',
        userId: 'test-user-1',
        assessmentType: 'comprehensive',
      });

      const dimensions = new Set(result.schema.sections.map((s) => s.riskDimension));

      // Should only contain valid dimensions
      const validDimensions = [
        'clinical_risk',
        'privacy_risk',
        'security_risk',
        'technical_credibility',
        'vendor_capability',
        'ai_transparency',
        'ethical_considerations',
        'regulatory_compliance',
        'operational_excellence',
        'sustainability',
      ];

      for (const dim of dimensions) {
        expect(validDimensions).toContain(dim);
      }
    });
  });

  describe('deterministic rendering', () => {
    it('same schema always produces same markdown', () => {
      const schema = fixtureQuestionnaireSchema();

      const markdown1 = questionnaireToMarkdown(schema);
      const markdown2 = questionnaireToMarkdown(schema);
      const markdown3 = questionnaireToMarkdown(schema);

      expect(markdown1).toBe(markdown2);
      expect(markdown2).toBe(markdown3);
    });
  });
});
```
