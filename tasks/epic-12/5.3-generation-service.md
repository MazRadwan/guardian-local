## Objective

Create the `QuestionnaireGenerationService` that orchestrates questionnaire generation with a **single Claude call**, returning both the JSON schema and pre-rendered markdown.

## Architecture

```
QuestionnaireGenerationService
├── Depends on: IClaudeClient (interface, not implementation)
├── Depends on: IQuestionRepository (for question persistence)
├── Depends on: AssessmentService (for assessment creation)
├── Depends on: VendorService (for vendor lookup)
├── Uses: buildQuestionGenerationPrompt() from questionGeneration.ts
├── Uses: questionnaireToMarkdown() from renderer
├── Uses: schemaToQuestions() adapter from 5.1 for persistence
└── Returns: { schema, assessmentId, markdown }
```

**Clean Architecture Compliance:**
- Lives in Application layer (`src/application/services/`)
- Depends on interfaces, not concrete implementations
- Uses `IQuestionRepository` directly for bulk insert (not `QuestionService`)
- Orchestrates but doesn't know about WebSocket/HTTP

**Note on QuestionService:**
The existing `QuestionService` has overlapping responsibility. See Part E below for deprecation path.

```typescript
/**
 * IQuestionnaireGenerationService - Interface for questionnaire generation
 *
 * Part of Epic 12.5: Hybrid Questionnaire Generation Architecture
 */

import { QuestionnaireSchema } from '../../domain/types/QuestionnaireSchema.js';

/**
 * Context for questionnaire generation
 */
export interface GenerationContext {
  /** Conversation this generation belongs to */
  conversationId: string;

  /** User requesting generation */
  userId: string;

  /** Type of assessment to generate */
  assessmentType: 'quick' | 'comprehensive' | 'category_focused';

  /** Vendor name (from conversation context) */
  vendorName?: string;

  /** Solution/product name */
  solutionName?: string;

  /** Summary of conversation context */
  contextSummary?: string;

  /** Focus categories for category_focused assessments */
  selectedCategories?: string[];
}

/**
 * Result of questionnaire generation
 */
export interface GenerationResult {
  /** The canonical questionnaire schema */
  schema: QuestionnaireSchema;

  /** Assessment ID (created or existing) */
  assessmentId: string;

  /** Pre-rendered markdown for chat display */
  markdown: string;
}

/**
 * Interface for questionnaire generation service
 */
export interface IQuestionnaireGenerationService {
  /**
   * Generate a questionnaire
   *
   * Makes a single Claude call, parses the JSON response,
   * persists to database, and returns schema + markdown.
   *
   * @param context - Generation context from conversation
   * @returns Schema, assessment ID, and pre-rendered markdown
   * @throws Error if Claude returns invalid JSON or generation fails
   */
  generate(context: GenerationContext): Promise<GenerationResult>;
}
```

### Part B: Create Service Implementation

**Path:** `packages/backend/src/application/services/QuestionnaireGenerationService.ts`

```typescript
/**
 * QuestionnaireGenerationService - Orchestrates questionnaire generation
 *
 * Part of Epic 12.5: Hybrid Questionnaire Generation Architecture
 *
 * This service:
 * 1. Builds the generation prompt
 * 2. Makes a single Claude call
 * 3. Parses and validates the JSON response
 * 4. Persists questions to the assessment
 * 5. Renders markdown for chat display
 *
 * Single source of truth: The JSON schema drives everything.
 */

import {
  IQuestionnaireGenerationService,
  GenerationContext,
  GenerationResult,
} from '../interfaces/IQuestionnaireGenerationService.js';
import {
  QuestionnaireSchema,
  validateQuestionnaireSchema,
  QUESTION_COUNT_RANGES,
} from '../../domain/types/QuestionnaireSchema.js';
import { IClaudeClient } from '../interfaces/IClaudeClient.js';
import { IQuestionRepository } from '../interfaces/IQuestionRepository.js';
import { AssessmentService } from './AssessmentService.js';
import { VendorService } from './VendorService.js';
import { ConversationService } from './ConversationService.js';
import { buildQuestionGenerationPrompt } from '../../infrastructure/ai/prompts/questionGeneration.js';
import { questionnaireToMarkdown } from '../../infrastructure/rendering/questionnaireToMarkdown.js';
import { schemaToQuestions } from '../adapters/QuestionnaireSchemaAdapter.js';

export class QuestionnaireGenerationService implements IQuestionnaireGenerationService {
  constructor(
    private readonly claudeClient: IClaudeClient,
    private readonly questionRepository: IQuestionRepository,
    private readonly assessmentService: AssessmentService,
    private readonly vendorService: VendorService,
    private readonly conversationService: ConversationService
  ) {}

  async generate(context: GenerationContext): Promise<GenerationResult> {
    console.log('[QuestionnaireGenerationService] Starting generation:', {
      conversationId: context.conversationId,
      assessmentType: context.assessmentType,
    });

    // 1. Build the generation prompt
    const prompt = buildQuestionGenerationPrompt({
      vendorType: context.vendorName || 'AI Vendor',
      solutionType: context.solutionName || 'AI Solution',
      assessmentFocus: context.contextSummary,
      assessmentType: context.assessmentType,
      category: context.selectedCategories?.[0],
    });

    // 2. Call Claude (single call, no streaming needed)
    const response = await this.claudeClient.sendMessage(
      [{ role: 'user', content: prompt }],
      {
        systemPrompt: 'You are a healthcare AI governance expert. Generate questionnaires in valid JSON format only. Do not include any text outside the JSON object.',
        // No tools - we want raw JSON output
      }
    );

    // 3. Parse JSON from response
    const schema = this.parseSchemaFromResponse(response.content);

    // 4. Validate schema
    if (!validateQuestionnaireSchema(schema)) {
      throw new Error('Claude returned invalid questionnaire schema');
    }

    // 5. Validate question count
    const range = QUESTION_COUNT_RANGES[context.assessmentType];
    if (schema.metadata.questionCount < range.min * 0.8 || schema.metadata.questionCount > range.max * 1.2) {
      console.warn(
        `[QuestionnaireGenerationService] Question count ${schema.metadata.questionCount} outside expected range [${range.min}, ${range.max}]`
      );
      // Don't throw - just warn. Claude may have good reasons.
    }

    // 6. Get or create assessment
    const assessmentId = await this.ensureAssessment(context, schema);

    // 7. Persist questions to assessment
    await this.persistQuestions(assessmentId, schema);

    // 8. Render markdown
    const markdown = questionnaireToMarkdown(schema);

    console.log('[QuestionnaireGenerationService] Generation complete:', {
      assessmentId,
      questionCount: schema.metadata.questionCount,
      sectionCount: schema.sections.length,
    });

    return {
      schema,
      assessmentId,
      markdown,
    };
  }

  /**
   * Parse QuestionnaireSchema from Claude's response
   */
  private parseSchemaFromResponse(content: string): QuestionnaireSchema {
    // Try to extract JSON from response
    let jsonString = content.trim();

    // Handle markdown code blocks
    const jsonMatch = content.match(/```(?:json)?\s*([\s\S]*?)```/);
    if (jsonMatch) {
      jsonString = jsonMatch[1].trim();
    }

    // Handle case where Claude adds explanation before/after JSON
    const objectMatch = jsonString.match(/\{[\s\S]*\}/);
    if (objectMatch) {
      jsonString = objectMatch[0];
    }

    try {
      const parsed = JSON.parse(jsonString);
      return parsed as QuestionnaireSchema;
    } catch (error) {
      console.error('[QuestionnaireGenerationService] Failed to parse JSON:', {
        error,
        contentPreview: content.substring(0, 500),
      });
      throw new Error('Failed to parse questionnaire JSON from Claude response');
    }
  }

  /**
   * Ensure assessment exists, create if needed
   */
  private async ensureAssessment(
    context: GenerationContext,
    schema: QuestionnaireSchema
  ): Promise<string> {
    // Check if conversation already has an assessment
    const conversation = await this.conversationService.getConversation(context.conversationId);

    if (conversation?.assessmentId) {
      return conversation.assessmentId;
    }

    // Get or create vendor
    const vendor = await this.vendorService.findOrCreateDefault(context.userId);

    // Create new assessment
    const assessment = await this.assessmentService.createAssessment({
      vendorName: schema.metadata.vendorName || vendor.name,
      assessmentType: context.assessmentType === 'category_focused' ? 'comprehensive' : context.assessmentType,
      solutionName: schema.metadata.solutionName || 'Assessment from Chat',
      createdBy: context.userId,
    });

    // Link to conversation
    await this.conversationService.linkAssessment(context.conversationId, assessment.assessmentId);

    return assessment.assessmentId;
  }

  /**
   * Persist questions from schema to assessment
   *
   * Uses the QuestionnaireSchemaAdapter from Story 5.1 to map:
   * - Schema question types (text, yes_no, scale, multiple_choice)
   *   → Entity question types (text, boolean, enum)
   * - Risk dimensions → Section numbers (1-10)
   *
   * Uses IQuestionRepository directly for bulk insert (not QuestionService).
   */
  private async persistQuestions(
    assessmentId: string,
    schema: QuestionnaireSchema
  ): Promise<void> {
    // Use adapter to convert schema to Question entities
    // This handles type mapping (yes_no → boolean, scale → enum, etc.)
    const questions = schemaToQuestions(schema, assessmentId);

    // Bulk insert questions using the repository directly
    await this.questionRepository.bulkInsert(questions);
  }
}
```

### Part D: Unit Tests

**Path:** `packages/backend/__tests__/unit/QuestionnaireGenerationService.test.ts`

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest';
import { QuestionnaireGenerationService } from '../../src/application/services/QuestionnaireGenerationService.js';
import { QuestionnaireSchema } from '../../src/domain/types/QuestionnaireSchema.js';

// Mock dependencies
const mockClaudeClient = {
  sendMessage: vi.fn(),
  streamMessage: vi.fn(),
};

const mockQuestionRepository = {
  bulkInsert: vi.fn(),
  findByAssessmentId: vi.fn(),
};

const mockAssessmentService = {
  createAssessment: vi.fn(),
  getAssessment: vi.fn(),
};

const mockVendorService = {
  findOrCreateDefault: vi.fn(),
};

const mockConversationService = {
  getConversation: vi.fn(),
  linkAssessment: vi.fn(),
};

/**
 * Create a valid schema fixture
 */
function createValidSchema(): QuestionnaireSchema {
  return {
    version: '1.0',
    metadata: {
      assessmentType: 'comprehensive',
      vendorName: 'Test Vendor',
      solutionName: 'Test Solution',
      generatedAt: new Date().toISOString(),
      questionCount: 90,
    },
    sections: [
      {
        id: 'privacy_risk',
        title: 'Privacy Risk',
        riskDimension: 'privacy_risk',
        description: 'Assess privacy practices.',
        questions: [
          {
            id: 'privacy_1',
            text: 'How do you handle PHI?',
            category: 'Data Handling',
            riskDimension: 'privacy_risk',
            questionType: 'text',
            required: true,
          },
        ],
      },
    ],
  };
}

describe('QuestionnaireGenerationService', () => {
  let service: QuestionnaireGenerationService;

  beforeEach(() => {
    vi.clearAllMocks();

    service = new QuestionnaireGenerationService(
      mockClaudeClient as any,
      mockQuestionRepository as any,
      mockAssessmentService as any,
      mockVendorService as any,
      mockConversationService as any
    );

    // Default mock responses
    mockConversationService.getConversation.mockResolvedValue({ assessmentId: null });
    mockVendorService.findOrCreateDefault.mockResolvedValue({ id: 'vendor-1', name: 'Default Vendor' });
    mockAssessmentService.createAssessment.mockResolvedValue({ assessmentId: 'assessment-123' });
    mockQuestionRepository.bulkInsert.mockResolvedValue(undefined);
    mockConversationService.linkAssessment.mockResolvedValue(undefined);
  });

  describe('generate', () => {
    it('returns schema, assessmentId, and markdown', async () => {
      const schema = createValidSchema();
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: JSON.stringify(schema),
      });

      const result = await service.generate({
        conversationId: 'conv-1',
        userId: 'user-1',
        assessmentType: 'comprehensive',
      });

      expect(result.schema).toEqual(schema);
      expect(result.assessmentId).toBe('assessment-123');
      expect(result.markdown).toContain('# Test Vendor Assessment Questionnaire');
    });

    it('parses JSON from markdown code block', async () => {
      const schema = createValidSchema();
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: '```json\n' + JSON.stringify(schema) + '\n```',
      });

      const result = await service.generate({
        conversationId: 'conv-1',
        userId: 'user-1',
        assessmentType: 'comprehensive',
      });

      expect(result.schema.version).toBe('1.0');
    });

    it('extracts JSON from mixed response', async () => {
      const schema = createValidSchema();
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: 'Here is your questionnaire:\n' + JSON.stringify(schema) + '\nLet me know if you need changes.',
      });

      const result = await service.generate({
        conversationId: 'conv-1',
        userId: 'user-1',
        assessmentType: 'comprehensive',
      });

      expect(result.schema.version).toBe('1.0');
    });

    it('throws on invalid JSON', async () => {
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: 'This is not valid JSON',
      });

      await expect(
        service.generate({
          conversationId: 'conv-1',
          userId: 'user-1',
          assessmentType: 'comprehensive',
        })
      ).rejects.toThrow('Failed to parse questionnaire JSON');
    });

    it('throws on invalid schema structure', async () => {
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: JSON.stringify({ invalid: 'schema' }),
      });

      await expect(
        service.generate({
          conversationId: 'conv-1',
          userId: 'user-1',
          assessmentType: 'comprehensive',
        })
      ).rejects.toThrow('invalid questionnaire schema');
    });

    it('reuses existing assessment if conversation has one', async () => {
      const schema = createValidSchema();
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: JSON.stringify(schema),
      });
      mockConversationService.getConversation.mockResolvedValue({
        assessmentId: 'existing-assessment',
      });

      const result = await service.generate({
        conversationId: 'conv-1',
        userId: 'user-1',
        assessmentType: 'comprehensive',
      });

      expect(result.assessmentId).toBe('existing-assessment');
      expect(mockAssessmentService.createAssessment).not.toHaveBeenCalled();
    });

    it('persists questions via repository bulkInsert', async () => {
      const schema = createValidSchema();
      mockClaudeClient.sendMessage.mockResolvedValue({
        content: JSON.stringify(schema),
      });

      await service.generate({
        conversationId: 'conv-1',
        userId: 'user-1',
        assessmentType: 'comprehensive',
      });

      // Uses IQuestionRepository.bulkInsert, not QuestionService
      expect(mockQuestionRepository.bulkInsert).toHaveBeenCalledWith(
        expect.arrayContaining([
          expect.objectContaining({
            questionText: 'How do you handle PHI?',
            sectionNumber: 2, // privacy_risk maps to section 2
          }),
        ])
      );
    });
  });
});
```
