### Step 2: Add Event Handler Method

Find the `WebSocketClient` class and add these methods:

```typescript
/**
 * Listen for questionnaire_ready event
 * Called when Claude determines it's ready to generate
 *
 * @param callback - Function to call with payload
 * @returns Unsubscribe function
 */
onQuestionnaireReady(
  callback: (data: QuestionnaireReadyPayload) => void
): () => void {
  if (!this.socket) {
    throw new Error('WebSocket not initialized');
  }

  const handler = (data: QuestionnaireReadyPayload) => {
    console.log('[WebSocket] Questionnaire ready:', data.conversationId);
    callback(data);
  };

  this.socket.on('questionnaire_ready', handler);

  // Return unsubscribe function
  return () => {
    this.socket?.off('questionnaire_ready', handler);
  };
}

/**
 * Emit generate_questionnaire event to trigger generation
 * Called when user clicks the "Generate" button
 *
 * @param conversationId - The conversation to generate for
 */
generateQuestionnaire(conversationId: string): void {
  if (!this.socket || !this.socket.connected) {
    throw new Error('WebSocket not connected');
  }

  console.log('[WebSocket] Requesting questionnaire generation:', conversationId);

  this.socket.emit('generate_questionnaire', {
    conversationId,
  } as GenerateQuestionnairePayload);
}
```

```typescript
// Add to imports
import { QuestionnaireReadyPayload } from '@/lib/websocket';

/**
 * Abstracted telemetry logging (future: integrate Sentry)
 */
const logTelemetry = (event: string, data: Record<string, unknown>) => {
  console.log(`[Telemetry] ${event}:`, data);
  // Future: Sentry.captureMessage(event, { extra: data });
};

// Inside the useEffect hook, add this handler registration:
// IMPORTANT: Ensure dependencies include wsClient and activeConversationId
// to prevent stale closures

useEffect(() => {
  if (!wsClient || !activeConversationId) return;

  // Handle questionnaire_ready event
  const unsubQuestionnaireReady = wsClient.onQuestionnaireReady((data) => {
    // Only process if for the active conversation
    if (data.conversationId === activeConversationId) {
      logTelemetry('questionnaire_ready_received', {
        conversationId: data.conversationId,
        assessmentType: data.assessmentType,
      });

      // Update store with pending questionnaire
      useChatStore.getState().setPendingQuestionnaire(data);
    }
  });

  // Cleanup: unsubscribe previous listener before re-registering
  return () => {
    unsubQuestionnaireReady();
  };
}, [wsClient, activeConversationId]); // Both deps required to prevent stale closures
```
