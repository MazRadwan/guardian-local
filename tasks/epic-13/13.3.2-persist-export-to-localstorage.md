# Story 13.3.2 - Persist Export State to localStorage

## Status: Not Started

## Context
The `exportReadyByConversation` cache in chatStore is in-memory only. After a page reload, export state is lost and users cannot download their questionnaire without regenerating. This story adds localStorage persistence for export state, similar to how payloads are persisted.

## Goal
Persist export state to localStorage so downloads survive page reload, not just conversation switches.

## Files to Modify
1. `apps/web/src/hooks/useQuestionnairePersistence.ts` - Add export persistence methods
2. `apps/web/src/hooks/useWebSocketEvents.ts` - Save export on `export_ready` event
3. `apps/web/src/components/chat/ChatInterface.tsx` - Load export on rehydration

## Required Changes

### 1. Add Export Persistence Methods to useQuestionnairePersistence.ts

```typescript
const EXPORT_KEY_PREFIX = 'guardian_q_export_';

// Add to returned object:
saveExport: (conversationId: string, payload: ExportReadyPayload) => {
  try {
    window.localStorage.setItem(
      `${EXPORT_KEY_PREFIX}${userId}:${conversationId}`,
      JSON.stringify(payload)
    );
  } catch (e) {
    console.error('Failed to save export:', e);
  }
},

loadExport: (conversationId: string): ExportReadyPayload | null => {
  try {
    const data = window.localStorage.getItem(
      `${EXPORT_KEY_PREFIX}${userId}:${conversationId}`
    );
    if (!data) return null;

    const parsed = JSON.parse(data);
    // Shape validation: conversationId match + required fields
    if (parsed.conversationId === conversationId &&
        parsed.assessmentId &&
        Array.isArray(parsed.formats)) {
      return parsed;
    }
    // Malformed entry - clear to prevent state poisoning
    window.localStorage.removeItem(`${EXPORT_KEY_PREFIX}${userId}:${conversationId}`);
    return null;
  } catch (e) {
    console.error('Failed to load export:', e);
    return null;
  }
},

clearExport: (conversationId: string) => {
  try {
    window.localStorage.removeItem(
      `${EXPORT_KEY_PREFIX}${userId}:${conversationId}`
    );
  } catch (e) {
    console.error('Failed to clear export:', e);
  }
},
```

### 1.5. Update `clearAllForUser` Method
Add export key prefix to the existing cleanup method:

```typescript
clearAllForUser: () => {
  try {
    const keysToRemove: string[] = [];
    for (let i = 0; i < window.localStorage.length; i++) {
      const key = window.localStorage.key(i);
      if (key && (
        key.startsWith(`${PAYLOAD_KEY_PREFIX}${userId}:`) ||
        key.startsWith(`${DISMISSED_KEY_PREFIX}${userId}:`) ||
        key.startsWith(`${EXPORT_KEY_PREFIX}${userId}:`)  // NEW: Clear export keys too
      )) {
        keysToRemove.push(key);
      }
    }
    keysToRemove.forEach(key => window.localStorage.removeItem(key));
  } catch (e) {
    console.error('Failed to clear all for user:', e);
  }
},
```

### 4. Update Rehydration in ChatInterface.tsx

Modify the rehydration effect to check localStorage export. Key implementation notes:
- **Use `useChatStore.getState()`** to access `setExportReady` since it's not destructured from the hook
- **Use existing `prevConversationIdRef` guard** to avoid repeated JSON parses on every render
- **Only run once per conversation change** - the guard handles this

```typescript
useEffect(() => {
  // GUARD: Only rehydrate on actual conversation change (prevents repeated JSON parses)
  if (prevConversationIdRef.current === activeConversationId) return;
  prevConversationIdRef.current = activeConversationId;

  // GUARD: Skip if generation in progress
  const { isGeneratingQuestionnaire } = useChatStore.getState();
  if (isGeneratingQuestionnaire) return;

  // Clear previous state
  useChatStore.getState().clearPendingQuestionnaire();
  useChatStore.getState().setQuestionnaireUIState('hidden');

  // Priority 1: Check in-memory export state (conversation switch)
  const exportData = useChatStore.getState().exportReadyByConversation?.[activeConversationId];
  if (exportData && exportData.conversationId === activeConversationId) {
    // Also restore payload for card rendering
    const savedPayload = persistence.loadPayload(activeConversationId);
    if (savedPayload && savedPayload.conversationId === activeConversationId && savedPayload.assessmentType) {
      useChatStore.getState().setPendingQuestionnaire(savedPayload);
    }
    useChatStore.getState().setQuestionnaireUIState('download');
    return;
  }

  // Priority 1.5: Check localStorage export (page reload case)
  const savedExport = persistence.loadExport(activeConversationId);
  if (savedExport && savedExport.conversationId === activeConversationId) {
    // Restore to in-memory cache via useChatStore.getState()
    useChatStore.getState().setExportReady(activeConversationId, savedExport);
    // Also restore payload for card rendering
    const savedPayload = persistence.loadPayload(activeConversationId);
    if (savedPayload && savedPayload.conversationId === activeConversationId && savedPayload.assessmentType) {
      useChatStore.getState().setPendingQuestionnaire(savedPayload);
    }
    useChatStore.getState().setQuestionnaireUIState('download');
    return;
  }

  // Priority 2: Check localStorage payload (ready state)
  const savedPayload = persistence.loadPayload(activeConversationId);
  if (savedPayload) {
    if (savedPayload.conversationId === activeConversationId && savedPayload.assessmentType) {
      useChatStore.getState().setPendingQuestionnaire(savedPayload);
      useChatStore.getState().setQuestionnaireUIState('ready');
    } else {
      // Malformed payload - cleanup
      persistence.clearPayload(activeConversationId);
    }
  }
}, [activeConversationId, user?.id]);
```

## State Priority Order (After Change)

| Priority | Source | Result State | When |
|----------|--------|--------------|------|
| 1 | In-memory `exportReadyByConversation` | 'download' | Conversation switch |
| 1.5 | localStorage export | 'download' | Page reload |
| 2 | localStorage payload | 'ready' | questionnaire_ready, no export |
| 3 | Neither | 'hidden' | No questionnaire activity |

## Security Considerations

- **User-scoped keys**: `guardian_q_export_{userId}:{conversationId}` prevents cross-user leaks
- **Shape validation**: Check conversationId match, assessmentId, and formats array
- **Malformed cleanup**: Remove invalid entries automatically
- **try/catch wrapping**: All localStorage operations wrapped for quota/permission errors

## Acceptance Criteria
- [ ] `saveExport`, `loadExport`, `clearExport` methods added to persistence hook
- [ ] Export state persisted to localStorage on `export_ready` event
- [ ] Page refresh restores download state (not just conversation switch)
- [ ] Export data validated before use (conversationId match, assessmentId, formats array)
- [ ] **Malformed entries cleared** - poisoned localStorage cleaned up automatically
- [ ] User-scoped keys prevent cross-user data leaks
- [ ] **Fresh event wins** - export_ready event overwrites both store AND localStorage
- [ ] No TypeScript errors

## Security & Validation Guards

| Guard | Location | Purpose |
|-------|----------|---------|
| **Conversation match** | loadExport | Verify export.conversationId === requested conversationId |
| **Shape validation** | loadExport | Check required fields: assessmentId, formats array |
| **Malformed cleanup** | loadExport | Auto-clear poisoned localStorage entries |
| **User-scoped keys** | saveExport/loadExport | Keys include userId to prevent cross-user leaks |
| **Fresh event wins** | handleExportReady | export_ready overwrites both store AND localStorage |
| **Active-convo gating** | rehydration effect | Only restore for activeConversationId |

describe('useQuestionnairePersistence - export methods', () => {
  const userId = 'user-123';
  const conversationId = 'conv-456';
  const validExport = {
    conversationId: 'conv-456',
    assessmentId: 'assess-789',
    formats: ['pdf', 'word', 'excel'],
    questionCount: 40,
  };

  beforeEach(() => {
    localStorage.clear();
  });

  describe('saveExport', () => {
    it('should save export to localStorage with user-scoped key', () => {
      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      result.current.saveExport(conversationId, validExport);

      const stored = localStorage.getItem(`guardian_q_export_${userId}:${conversationId}`);
      expect(JSON.parse(stored!)).toEqual(validExport);
    });
  });

  describe('loadExport', () => {
    it('should return null when no export exists', () => {
      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      expect(result.current.loadExport(conversationId)).toBeNull();
    });

    it('should return valid export when present', () => {
      localStorage.setItem(
        `guardian_q_export_${userId}:${conversationId}`,
        JSON.stringify(validExport)
      );

      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      expect(result.current.loadExport(conversationId)).toEqual(validExport);
    });

    it('should return null and clear malformed export (missing assessmentId)', () => {
      const malformed = { conversationId, formats: ['pdf'] }; // missing assessmentId
      localStorage.setItem(
        `guardian_q_export_${userId}:${conversationId}`,
        JSON.stringify(malformed)
      );

      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      expect(result.current.loadExport(conversationId)).toBeNull();
      expect(localStorage.getItem(`guardian_q_export_${userId}:${conversationId}`)).toBeNull();
    });

    it('should return null and clear malformed export (wrong conversationId)', () => {
      const malformed = { ...validExport, conversationId: 'wrong-id' };
      localStorage.setItem(
        `guardian_q_export_${userId}:${conversationId}`,
        JSON.stringify(malformed)
      );

      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      expect(result.current.loadExport(conversationId)).toBeNull();
      expect(localStorage.getItem(`guardian_q_export_${userId}:${conversationId}`)).toBeNull();
    });

    it('should return null and clear malformed export (formats not array)', () => {
      const malformed = { ...validExport, formats: 'pdf' }; // not an array
      localStorage.setItem(
        `guardian_q_export_${userId}:${conversationId}`,
        JSON.stringify(malformed)
      );

      const { result } = renderHook(() => useQuestionnairePersistence(userId));

      expect(result.current.loadExport(conversationId)).toBeNull();
    });
  });

  describe('clearExport', () => {
    it('should remove export from localStorage', () => {
      localStorage.setItem(
        `guardian_q_export_${userId}:${conversationId}`,
        JSON.stringify(validExport)
      );

      const { result } = renderHook(() => useQuestionnairePersistence(userId));
      result.current.clearExport(conversationId);

      expect(localStorage.getItem(`guardian_q_export_${userId}:${conversationId}`)).toBeNull();
    });
  });

  describe('clearAllForUser', () => {
    it('should clear export keys along with payload and dismissed keys', () => {
      // Set up various keys
      localStorage.setItem(`guardian_q_export_${userId}:conv-1`, JSON.stringify(validExport));
      localStorage.setItem(`guardian_q_payload_${userId}:conv-2`, JSON.stringify({}));
      localStorage.setItem(`guardian_dismissed_${userId}:conv-3`, 'true');
      localStorage.setItem('unrelated_key', 'should remain');

      const { result } = renderHook(() => useQuestionnairePersistence(userId));
      result.current.clearAllForUser();

      expect(localStorage.getItem(`guardian_q_export_${userId}:conv-1`)).toBeNull();
      expect(localStorage.getItem(`guardian_q_payload_${userId}:conv-2`)).toBeNull();
      expect(localStorage.getItem(`guardian_dismissed_${userId}:conv-3`)).toBeNull();
      expect(localStorage.getItem('unrelated_key')).toBe('should remain');
    });
  });
});
