# Story 13.5.5 - Add Backend Tests for Phase Events

## Status: Complete

## Context
Backend emits phase events during generation (13.5.2). This story adds tests to verify correct event emission.

## Goal
Test that `generation_phase` events are emitted correctly during questionnaire generation.

## Files to Create/Modify
- `packages/backend/__tests__/unit/ChatServer.handleGenerateQuestionnaire.test.ts` (update existing or create new)

## Implementation

### Test Suite
```typescript
import { ChatServer } from '../../src/infrastructure/websocket/ChatServer';
import { GENERATION_PHASES } from '@guardian/shared';

describe('ChatServer handleGenerateQuestionnaire', () => {
  let chatServer: ChatServer;
  let mockSocket: any;
  let mockQuestionnaireService: any;
  let emittedEvents: { event: string; data: any }[];

  beforeEach(() => {
    emittedEvents = [];
    mockSocket = {
      emit: jest.fn((event, data) => {
        emittedEvents.push({ event, data });
      }),
      id: 'test-socket-id',
    };
    mockQuestionnaireService = {
      generate: jest.fn().mockResolvedValue({
        schema: { /* mock schema */ },
        assessmentId: 'test-assessment-id',
        markdown: '# Test Questionnaire',
      }),
    };
    // Setup chatServer with mocks...
  });

  describe('phase events', () => {
    it('emits generation_phase events in order (0, 1, 2, 3)', async () => {
      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const phaseEvents = emittedEvents.filter(e => e.event === 'generation_phase');

      expect(phaseEvents).toHaveLength(4);
      expect(phaseEvents[0].data.phase).toBe(0);
      expect(phaseEvents[0].data.phaseId).toBe('context');
      expect(phaseEvents[1].data.phase).toBe(1);
      expect(phaseEvents[1].data.phaseId).toBe('generating');
      expect(phaseEvents[2].data.phase).toBe(2);
      expect(phaseEvents[2].data.phaseId).toBe('validating');
      expect(phaseEvents[3].data.phase).toBe(3);
      expect(phaseEvents[3].data.phaseId).toBe('saving');
    });

    it('emits export_ready after all phase events', async () => {
      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const eventNames = emittedEvents.map(e => e.event);
      const exportReadyIndex = eventNames.lastIndexOf('export_ready');
      const lastPhaseIndex = eventNames.lastIndexOf('generation_phase');

      expect(exportReadyIndex).toBeGreaterThan(lastPhaseIndex);
    });

    it('includes conversationId in all phase events', async () => {
      const conversationId = 'test-convo-123';

      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId,
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const phaseEvents = emittedEvents.filter(e => e.event === 'generation_phase');
      phaseEvents.forEach(e => {
        expect(e.data.conversationId).toBe(conversationId);
      });
    });

    it('includes timestamp in all phase events', async () => {
      const before = Date.now();

      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const after = Date.now();
      const phaseEvents = emittedEvents.filter(e => e.event === 'generation_phase');

      phaseEvents.forEach(e => {
        expect(e.data.timestamp).toBeGreaterThanOrEqual(before);
        expect(e.data.timestamp).toBeLessThanOrEqual(after);
      });
    });

    it('phaseId matches GENERATION_PHASES constant', async () => {
      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const phaseEvents = emittedEvents.filter(e => e.event === 'generation_phase');

      phaseEvents.forEach((e, idx) => {
        expect(e.data.phaseId).toBe(GENERATION_PHASES[idx]);
      });
    });
  });

  describe('error handling', () => {
    it('emits only phase 0 when Claude call fails', async () => {
      mockQuestionnaireService.generate.mockRejectedValue(new Error('Claude API error'));

      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const phaseEvents = emittedEvents.filter(e => e.event === 'generation_phase');

      // Phase 0 emitted before Claude call, then error
      expect(phaseEvents.length).toBeLessThanOrEqual(1);
      if (phaseEvents.length === 1) {
        expect(phaseEvents[0].data.phase).toBe(0);
      }
    });

    it('emits extraction_failed on error', async () => {
      mockQuestionnaireService.generate.mockRejectedValue(new Error('Generation failed'));

      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const errorEvents = emittedEvents.filter(e => e.event === 'extraction_failed');
      expect(errorEvents).toHaveLength(1);
    });

    it('does not emit export_ready on error', async () => {
      mockQuestionnaireService.generate.mockRejectedValue(new Error('Generation failed'));

      await chatServer.handleGenerateQuestionnaire(mockSocket, {
        conversationId: 'test-convo',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
      });

      const exportEvents = emittedEvents.filter(e => e.event === 'export_ready');
      expect(exportEvents).toHaveLength(0);
    });
  });
});
```
