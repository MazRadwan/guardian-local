### 3. Add Export State Restoration Tests

```typescript
describe('rehydration', () => {
  describe('export state restoration', () => {
    it('should restore download state if export exists for conversation', () => {
      // Setup: Set export in store
      useChatStore.setState({
        exportReadyByConversation: {
          'conv-123': {
            conversationId: 'conv-123',
            assessmentId: 'assess-456',
            formats: ['pdf', 'word', 'excel'],
            questionCount: 40,
          },
        },
      });

      // Mock: No localStorage payload
      mockPersistence.loadPayload.mockReturnValue(null);

      // Act: Render with conversation that has export
      render(<ChatInterface conversationId="conv-123" />);

      // Assert: UI state should be 'download'
      expect(useChatStore.getState().questionnaireUIState).toBe('download');
    });

    it('should restore ready state if payload exists but no export', () => {
      // Setup: No export in store
      useChatStore.setState({
        exportReadyByConversation: {},
      });

      // Mock: localStorage has payload
      const mockPayload = {
        conversationId: 'conv-123',
        assessmentType: 'quick',
        vendorName: 'Test Vendor',
        // ... other fields
      };
      mockPersistence.loadPayload.mockReturnValue(mockPayload);

      // Act: Render with conversation that has payload
      render(<ChatInterface conversationId="conv-123" />);

      // Assert: UI state should be 'ready', pendingQuestionnaire set
      expect(useChatStore.getState().questionnaireUIState).toBe('ready');
      expect(useChatStore.getState().pendingQuestionnaire).toEqual(mockPayload);
    });

    it('should show hidden state if no payload and no export', () => {
      // Setup: No export in store
      useChatStore.setState({
        exportReadyByConversation: {},
      });

      // Mock: No localStorage payload
      mockPersistence.loadPayload.mockReturnValue(null);

      // Act: Render with fresh conversation
      render(<ChatInterface conversationId="conv-123" />);

      // Assert: UI state should be 'hidden'
      expect(useChatStore.getState().questionnaireUIState).toBe('hidden');
    });

    it('should prioritize export over payload when both exist', () => {
      // Setup: Both export AND payload exist
      useChatStore.setState({
        exportReadyByConversation: {
          'conv-123': {
            conversationId: 'conv-123',
            assessmentId: 'assess-456',
            formats: ['pdf'],
            questionCount: 40,
          },
        },
      });

      mockPersistence.loadPayload.mockReturnValue({
        conversationId: 'conv-123',
        assessmentType: 'quick',
        // ...
      });

      // Act: Render
      render(<ChatInterface conversationId="conv-123" />);

      // Assert: Export wins - UI state should be 'download'
      expect(useChatStore.getState().questionnaireUIState).toBe('download');
      // loadPayload should NOT be called since export found first
      expect(mockPersistence.loadPayload).not.toHaveBeenCalled();
    });
  });
});
```
