### VerticalStepper.test.tsx (NEW)

```typescript
import { render, screen } from '@testing-library/react';
import { VerticalStepper, Step } from '../VerticalStepper';

const mockSteps: Step[] = [
  { id: 'context', label: 'Context gathered' },
  { id: 'generating', label: 'Generating questions' },
  { id: 'validating', label: 'Validating structure' },
  { id: 'saving', label: 'Saving assessment' },
];

describe('VerticalStepper', () => {
  describe('rendering', () => {
    it('renders no steps when currentStep is -1', () => {
      render(
        <VerticalStepper steps={mockSteps} currentStep={-1} isRunning={false} />
      );

      expect(screen.queryByText('Context gathered')).not.toBeInTheDocument();
    });

    it('renders steps up to currentStep (inclusive)', () => {
      render(
        <VerticalStepper steps={mockSteps} currentStep={1} isRunning={true} />
      );

      expect(screen.getByText('Context gathered')).toBeInTheDocument();
      expect(screen.getByText(/Generating questions/)).toBeInTheDocument();
      expect(screen.queryByText('Validating structure')).not.toBeInTheDocument();
    });

    it('renders all steps when currentStep equals length', () => {
      render(
        <VerticalStepper steps={mockSteps} currentStep={4} isRunning={false} />
      );

      expect(screen.getByText('Context gathered')).toBeInTheDocument();
      expect(screen.getByText('Generating questions')).toBeInTheDocument();
      expect(screen.getByText('Validating structure')).toBeInTheDocument();
      expect(screen.getByText('Saving assessment')).toBeInTheDocument();
    });
  });

  describe('step indicators', () => {
    it('shows checkmark for completed steps', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={2} isRunning={true} />
      );

      // First two steps should have checkmarks (svg with check path)
      const checkmarks = container.querySelectorAll('svg path[d*="M5 13l4 4L19 7"]');
      expect(checkmarks).toHaveLength(2);
    });

    it('shows pulse indicator for active step', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={1} isRunning={true} />
      );

      // Active step should have pulsing dot
      const pulseDot = container.querySelector('.animate-pulse.rounded-full');
      expect(pulseDot).toBeInTheDocument();
    });

    it('shows "..." suffix on active step label', () => {
      render(
        <VerticalStepper steps={mockSteps} currentStep={1} isRunning={true} />
      );

      expect(screen.getByText(/Generating questions/)).toHaveTextContent('...');
    });
  });

  describe('connecting lines', () => {
    it('shows green line for completed steps', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={2} isRunning={true} />
      );

      const greenLines = container.querySelectorAll('.bg-emerald-300');
      expect(greenLines.length).toBeGreaterThan(0);
    });

    it('does not show line after last rendered step', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={0} isRunning={true} />
      );

      // Only one step rendered, should have no connecting line
      const lines = container.querySelectorAll('.min-h-\\[16px\\]');
      expect(lines).toHaveLength(0);
    });
  });

  describe('animations', () => {
    it('applies fadeIn animation class', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={1} isRunning={true} />
      );

      const animatedElements = container.querySelectorAll('.animate-fadeIn');
      expect(animatedElements.length).toBeGreaterThan(0);
    });

    it('applies staggered animation delay', () => {
      const { container } = render(
        <VerticalStepper steps={mockSteps} currentStep={2} isRunning={true} />
      );

      const steps = container.querySelectorAll('.animate-fadeIn');
      // Check that delays are different (100ms, 200ms, etc.)
      const delays = Array.from(steps).map(s => s.style.animationDelay);
      expect(delays[0]).toBe('0ms');
      expect(delays[1]).toBe('100ms');
      expect(delays[2]).toBe('200ms');
    });
  });
});
```

```typescript
describe('simulation gating', () => {

describe('collapse/expand behavior', () => {
  beforeEach(() => {
    jest.useFakeTimers();
    useChatStore.setState(mockStoreWithStepper);
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('auto-expands when generation starts', () => {
    const { rerender } = render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="ready"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Start generation
    useChatStore.setState({ currentGenerationStep: 0 });
    rerender(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="generating"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Stepper content should be visible (expanded)
    expect(screen.getByText('Context gathered')).toBeVisible();
  });

  it('auto-collapses 800ms after completion', async () => {
    useChatStore.setState({ currentGenerationStep: 4 }); // Complete

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="generating"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Initially expanded
    expect(screen.getByText('Context gathered')).toBeVisible();

    // Fast-forward 800ms
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Should be collapsed (stepper content hidden)
    expect(screen.getByText('Context gathered')).not.toBeVisible();
  });

  it('allows manual toggle via header click', async () => {
    useChatStore.setState({ currentGenerationStep: 1 });

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="generating"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    const header = screen.getByText('Generating Assessment').closest('button');

    // Click to collapse
    await userEvent.click(header!);
    expect(screen.getByText('Context gathered')).not.toBeVisible();

    // Click to expand
    await userEvent.click(header!);
    expect(screen.getByText('Context gathered')).toBeVisible();
  });

  it('shows summary when collapsed and complete', async () => {
    useChatStore.setState({ currentGenerationStep: 4 }); // Complete

    render(
      <QuestionnairePromptCard
        payload={{ ...mockPayload, questionCount: 40, assessmentType: 'quick' }}
        uiState="generating"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Fast-forward to collapse
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Summary should be visible
    expect(screen.getByText(/40 questions/)).toBeInTheDocument();
  });
});
```

## Test Robustness Guidelines

- Use `GENERATION_STEPS` from store, not hardcoded step count
- Use fake timers (`jest.useFakeTimers()`) for animation/delay tests
- Don't rely on specific timing - use `jest.advanceTimersByTime()`
- Use `act()` wrapper for state updates that trigger re-renders
- Test conversation gating by mocking `useChatStore.getState()`
