```typescript
import { render, screen, act } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { QuestionnairePromptCard } from '../QuestionnairePromptCard';
import { useChatStore } from '@/stores/chatStore';
import { GENERATION_STEPS } from '@/constants/generation';  // Use constant, not hardcoded values

const mockPayload = {
  vendorName: 'Test Vendor',
  assessmentType: 'Quick Assessment',
  questionCount: 40,
};

const mockExportData = {
  formats: ['pdf', 'word', 'excel'],
  assessmentId: 'test-assessment-123',
};

describe('post-completion collapse behavior', () => {
  beforeEach(() => {
    jest.useFakeTimers();
    useChatStore.setState({
      currentGenerationStep: 4,  // Complete
      generationSteps: [
        { id: 'context', label: 'Context gathered' },
        { id: 'generating', label: 'Generating questions' },
        { id: 'validating', label: 'Validating structure' },
        { id: 'saving', label: 'Saving assessment' },
      ],
    });
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('auto-collapses 800ms after successful completion', () => {
    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Initially expanded - stepper content visible
    expect(screen.getByText('Context gathered')).toBeVisible();

    // Fast-forward 800ms
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Should be collapsed - stepper content hidden
    expect(screen.getByText('Context gathered')).not.toBeVisible();
  });

  it('does NOT auto-collapse on error state', () => {
    useChatStore.setState({ currentGenerationStep: 2 });  // Mid-generation

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="error"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Should still be expanded (showing where it failed)
    expect(screen.getByText('Context gathered')).toBeVisible();
  });

  it('does NOT auto-collapse if complete then error transition', () => {
    const { rerender } = render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Switch to error before timer fires
    rerender(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="error"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Timer should have been cleared, no collapse
    expect(screen.getByText('Context gathered')).toBeVisible();
  });
});

describe('conversation scoping', () => {
  beforeEach(() => {
    jest.useFakeTimers();
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('collapse state does not bleed across conversations', () => {
    useChatStore.setState({
      currentGenerationStep: GENERATION_STEPS.length,  // Use constant
      activeConversationId: 'convo-a',
    });

    const { rerender } = render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Collapse conversation A
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Switch to conversation B (reset step to simulate new convo)
    useChatStore.setState({
      currentGenerationStep: 0,
      activeConversationId: 'convo-b',
    });

    rerender(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="generating"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Conversation B should start expanded
    expect(screen.getByText('Context gathered')).toBeVisible();
  });

  it('clears pending collapse timer when conversation switches mid-delay', () => {
    useChatStore.setState({
      currentGenerationStep: GENERATION_STEPS.length,
      activeConversationId: 'convo-a',
    });

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Advance 400ms (halfway through collapse delay)
    act(() => {
      jest.advanceTimersByTime(400);
    });

    // Switch conversation mid-timer
    useChatStore.setState({ activeConversationId: 'convo-b' });

    // Advance remaining 400ms - timer should have been cleared
    act(() => {
      jest.advanceTimersByTime(400);
    });

    // Should still be expanded (timer was cleared on conversation switch)
    expect(screen.getByText('Context gathered')).toBeVisible();
  });
});

describe('download buttons', () => {
  beforeEach(() => {
    jest.useFakeTimers();
    useChatStore.setState({ currentGenerationStep: GENERATION_STEPS.length });
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('download buttons visible when collapsed', async () => {
    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    // Auto-collapse
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Download buttons should still be visible
    expect(screen.getByRole('button', { name: /word/i })).toBeVisible();
    expect(screen.getByRole('button', { name: /pdf/i })).toBeVisible();
    expect(screen.getByRole('button', { name: /excel/i })).toBeVisible();
  });

  it('download buttons work when collapsed', async () => {
    const onDownload = jest.fn();
    const user = userEvent.setup({ advanceTimers: jest.advanceTimersByTime });

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={onDownload}
        onRetry={jest.fn()}
      />
    );

    // Collapse
    act(() => {
      jest.advanceTimersByTime(800);
    });

    // Click download
    await user.click(screen.getByRole('button', { name: /word/i }));

    expect(onDownload).toHaveBeenCalledWith('word');
  });

  it('downloads do not clear export state (durability)', async () => {
    const user = userEvent.setup({ advanceTimers: jest.advanceTimersByTime });

    useChatStore.setState({
      currentGenerationStep: GENERATION_STEPS.length,
      exportReadyByConversation: { 'convo-1': mockExportData },
    });

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    await user.click(screen.getByRole('button', { name: /pdf/i }));

    // Export data should still be there
    expect(useChatStore.getState().exportReadyByConversation['convo-1']).toBeDefined();
  });

  it('allows multiple downloads', async () => {
    const onDownload = jest.fn();
    const user = userEvent.setup({ advanceTimers: jest.advanceTimersByTime });

    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={onDownload}
        onRetry={jest.fn()}
      />
    );

    await user.click(screen.getByRole('button', { name: /word/i }));
    await user.click(screen.getByRole('button', { name: /pdf/i }));
    await user.click(screen.getByRole('button', { name: /excel/i }));

    expect(onDownload).toHaveBeenCalledTimes(3);
    expect(onDownload).toHaveBeenNthCalledWith(1, 'word');
    expect(onDownload).toHaveBeenNthCalledWith(2, 'pdf');
    expect(onDownload).toHaveBeenNthCalledWith(3, 'excel');
  });
});

describe('summary fallbacks', () => {
  beforeEach(() => {
    jest.useFakeTimers();
    useChatStore.setState({ currentGenerationStep: GENERATION_STEPS.length });
  });

  afterEach(() => {
    jest.useRealTimers();
  });

  it('shows full summary when all data present', () => {
    render(
      <QuestionnairePromptCard
        payload={mockPayload}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    expect(screen.getByText(/40 questions/)).toBeInTheDocument();
    expect(screen.getByText(/Quick Assessment/)).toBeInTheDocument();
  });

  it('shows fallback when questionCount missing', () => {
    render(
      <QuestionnairePromptCard
        payload={{ ...mockPayload, questionCount: undefined }}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    expect(screen.getByText(/Assessment complete/)).toBeInTheDocument();
  });

  it('omits assessmentType segment when missing', () => {
    render(
      <QuestionnairePromptCard
        payload={{ ...mockPayload, assessmentType: undefined }}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    expect(screen.getByText(/40 questions/)).toBeInTheDocument();
    // Should not have the type
    expect(screen.queryByText(/Quick Assessment/)).not.toBeInTheDocument();
  });

  it('handles both fields missing gracefully', () => {
    render(
      <QuestionnairePromptCard
        payload={{ vendorName: 'Test', questionCount: undefined, assessmentType: undefined }}
        uiState="download"
        onGenerate={jest.fn()}
        onDismiss={jest.fn()}
        onDownload={jest.fn()}
        onRetry={jest.fn()}
      />
    );

    act(() => {
      jest.advanceTimersByTime(800);
    });

    expect(screen.getByText(/Assessment complete/)).toBeInTheDocument();
  });
});
```
