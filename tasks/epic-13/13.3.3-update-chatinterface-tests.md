describe('download durability', () => {
  it('should keep download state after successful download', async () => {

### 3. Add Export localStorage Restoration Tests

```typescript
describe('export localStorage restoration', () => {
  it('should restore export state from localStorage on page reload', () => {
    // Setup: localStorage has export data
    const savedExport = {
      conversationId: 'conv-123',
      assessmentId: 'assess-456',
      formats: ['pdf', 'word', 'excel'],
      questionCount: 40,
    };
    mockPersistence.loadExport.mockReturnValue(savedExport);

    // No in-memory export
    useChatStore.setState({
      exportReadyByConversation: {},
    });

    // Act: Mount component (simulates page reload)
    render(<ChatInterface conversationId="conv-123" />);

    // Assert: UI state is 'download', export restored to store
    expect(useChatStore.getState().questionnaireUIState).toBe('download');
    expect(useChatStore.getState().exportReadyByConversation['conv-123']).toEqual(savedExport);
  });

  it('should prioritize in-memory export over localStorage', () => {
    // Setup: Both in-memory and localStorage have export
    const inMemoryExport = {
      conversationId: 'conv-123',
      assessmentId: 'assess-memory',
      formats: ['pdf'],
      questionCount: 40,
    };
    const localStorageExport = {
      conversationId: 'conv-123',
      assessmentId: 'assess-storage',
      formats: ['word'],
      questionCount: 50,
    };

    useChatStore.setState({
      exportReadyByConversation: { 'conv-123': inMemoryExport },
    });
    mockPersistence.loadExport.mockReturnValue(localStorageExport);

    // Act: Mount
    render(<ChatInterface conversationId="conv-123" />);

    // Assert: In-memory wins (no loadExport call needed)
    expect(useChatStore.getState().exportReadyByConversation['conv-123']).toEqual(inMemoryExport);
  });

  it('should clear malformed localStorage export', () => {
    // Setup: Malformed export (wrong conversationId)
    const malformedExport = {
      conversationId: 'wrong-conv',
      assessmentId: 'assess-456',
      formats: ['pdf'],
      questionCount: 40,
    };
    mockPersistence.loadExport.mockReturnValue(malformedExport);

    useChatStore.setState({
      exportReadyByConversation: {},
    });

    // Act: Mount
    render(<ChatInterface conversationId="conv-123" />);

    // Assert: UI state is 'hidden' (malformed rejected)
    // Note: loadExport already validates and returns null for malformed
    expect(useChatStore.getState().questionnaireUIState).toBe('hidden');
  });

  // NEW: Active-conversation gating
  it('should not restore export for inactive conversation', () => {
    // Setup: localStorage has export for conv-456
    const otherConvoExport = {
      conversationId: 'conv-456',
      assessmentId: 'assess-789',
      formats: ['pdf'],
      questionCount: 30,
    };
    mockPersistence.loadExport.mockReturnValue(otherConvoExport);

    useChatStore.setState({
      exportReadyByConversation: {},
    });

    // Act: Mount component with activeConversationId = conv-123
    render(<ChatInterface conversationId="conv-123" />);

    // Assert: UI state is 'hidden' (not conv-456's state)
    expect(useChatStore.getState().questionnaireUIState).toBe('hidden');
  });

  // NEW: Fresh event wins
  it('should prioritize fresh export_ready event over localStorage', () => {
    // Setup: localStorage has old export
    const oldExport = {
      conversationId: 'conv-123',
      assessmentId: 'assess-old',
      formats: ['pdf'],
      questionCount: 30,
    };
    mockPersistence.loadExport.mockReturnValue(oldExport);

    // In-memory has fresh export (simulates event arriving after rehydrate)
    const freshExport = {
      conversationId: 'conv-123',
      assessmentId: 'assess-fresh',
      formats: ['pdf', 'word'],
      questionCount: 50,
    };
    useChatStore.setState({
      exportReadyByConversation: { 'conv-123': freshExport },
    });

    // Act: Mount
    render(<ChatInterface conversationId="conv-123" />);

    // Assert: Fresh export wins
    expect(useChatStore.getState().exportReadyByConversation['conv-123']).toEqual(freshExport);
    expect(useChatStore.getState().questionnaireUIState).toBe('download');
  });
});
```
