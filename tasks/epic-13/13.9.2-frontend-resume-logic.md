# Story 13.9.2 - Frontend: Request Export Status on Resume

## Status: Not Started

## Context
When a user returns to a conversation, the frontend should check if an export already exists. This story adds the client-side logic to request export status from the server on resume.

## Goal
On conversation activation, request export status from server if no cached data exists. Handle the response using existing `export_ready` flow.

## Files to Modify

### 1. `apps/web/src/lib/websocket.ts`

Add new types and methods to WebSocketClient:

```typescript
// Add to exports near other payload types
export interface ExportStatusNotFoundPayload {
  conversationId: string;
}

export interface ExportStatusErrorPayload {
  conversationId: string;
  error: string;
}

// Add to WebSocketClient class:

/**
 * Request export status for a conversation (13.9.2)
 */
requestExportStatus(conversationId: string): void {
  if (!this.socket?.connected) {
    console.warn('[WebSocket] Cannot request export status: not connected');
    return;
  }
  console.log('[WebSocket] Requesting export status for:', conversationId);
  this.socket.emit('get_export_status', { conversationId });
}

/**
 * Subscribe to export_status_not_found events (13.9.2)
 */
onExportStatusNotFound(
  callback: (data: ExportStatusNotFoundPayload) => void
): () => void {
  if (!this.socket) throw new Error('WebSocket not initialized');

  const handler = (data: ExportStatusNotFoundPayload) => {
    console.log('[WebSocket] Export status not found:', data.conversationId);
    callback(data);
  };

  this.socket.on('export_status_not_found', handler);
  return () => this.socket?.off('export_status_not_found', handler);
}

/**
 * Subscribe to export_status_error events (13.9.2)
 */
onExportStatusError(
  callback: (data: ExportStatusErrorPayload) => void
): () => void {
  if (!this.socket) throw new Error('WebSocket not initialized');

  const handler = (data: ExportStatusErrorPayload) => {
    console.error('[WebSocket] Export status error:', data);
    callback(data);
  };

  this.socket.on('export_status_error', handler);
  return () => this.socket?.off('export_status_error', handler);
}
```

### 2. `apps/web/src/hooks/useWebSocketEvents.ts`

Add resume logic after existing effect subscriptions:

**IMPORTANT:**
1. Use `socket?.connected` check before emitting (not just `socket` existence)
2. Check `questionnaireReadyPayload` guard - only request if questionnaire was ready
3. Dependency should be the **current conversation's export flag**, not the whole map
4. localStorage key must match 13.3 pattern exactly: `export_${conversationId}`

```typescript
import { useRef } from 'react';

// Inside useWebSocketEvents hook, add:

// Track pending export status requests to avoid duplicates
const pendingExportStatusRef = useRef<Set<string>>(new Set());

// Derive the current conversation's export data (not whole map, avoids unnecessary re-runs)
const currentExportData = activeConversationId
  ? exportReadyByConversation[activeConversationId]
  : undefined;

// Story 13.9.2: Request export status when conversation changes and no cached export
useEffect(() => {
  if (!wsClient || !activeConversationId) return;

  // Short-circuit if not connected
  if (!wsClient.isConnected?.()) {
    console.log('[useWebSocketEvents] WebSocket not connected, skipping export status request');
    return;
  }

  // Check if we already have export data (in-memory)
  if (currentExportData) {
    console.log('[useWebSocketEvents] Export data cached in memory, skipping server request');
    return;
  }

  // Check localStorage (key must match 13.3 pattern exactly)
  const storageKey = `export_${activeConversationId}`;
  const stored = localStorage.getItem(storageKey);
  if (stored) {
    try {
      const parsed = JSON.parse(stored);
      console.log('[useWebSocketEvents] Export data found in localStorage, restoring');
      useChatStore.getState().setExportData(activeConversationId, parsed);
      useChatStore.getState().setQuestionnaireUIState('download');
      return;
    } catch (e) {
      console.warn('[useWebSocketEvents] Invalid localStorage export data, continuing to server');
      localStorage.removeItem(storageKey);  // Clean up corrupted data
    }
  }

  // NOTE: Do NOT guard on questionnaireReadyPayload here.
  // On fresh load (new device/tab), that payload is null but user may have
  // generated a questionnaire in a previous session. Let server respond with
  // export_status_not_found to control spam - that's the authoritative source.

  // Avoid duplicate requests for same conversation
  if (pendingExportStatusRef.current.has(activeConversationId)) {
    console.log('[useWebSocketEvents] Export status request already pending');
    return;
  }

  // Request from server
  console.log('[useWebSocketEvents] Requesting export status from server');
  pendingExportStatusRef.current.add(activeConversationId);
  wsClient.requestExportStatus(activeConversationId);

}, [wsClient, activeConversationId, currentExportData]);  // Only currentExportData, not whole map

// Story 13.9.2: Handle export_status_not_found
useEffect(() => {
  if (!wsClient) return;

  const unsubscribe = wsClient.onExportStatusNotFound((data) => {
    pendingExportStatusRef.current.delete(data.conversationId);
    console.log('[useWebSocketEvents] No export found for conversation, user can generate');
    // UI remains in ready state - no action needed
  });

  return unsubscribe;
}, [wsClient]);

// Story 13.9.2: Handle export_status_error
useEffect(() => {
  if (!wsClient) return;

  const unsubscribe = wsClient.onExportStatusError((data) => {
    pendingExportStatusRef.current.delete(data.conversationId);
    console.error('[useWebSocketEvents] Export status error:', data.error);
    // Don't disrupt UX - user can still generate if needed
  });

  return unsubscribe;
}, [wsClient]);
```

**Note:** The existing `onExportReady` handler already handles `export_ready` events from the server, so no changes needed there. The server reuses that event.

### 3. Add tests to existing `apps/web/src/hooks/__tests__/useWebSocketEvents.test.ts`

```typescript
// Add to existing describe block or create new describe section:

describe('export status resume flow (Story 13.9.2)', () => {
  // Extend mockWsClient with new methods if not already present:
  const mockWsClient = {
    // ... existing mocks
    requestExportStatus: jest.fn(),
    onExportStatusNotFound: jest.fn(() => jest.fn()),
    onExportStatusError: jest.fn(() => jest.fn()),
    isConnected: jest.fn(() => true),
  };

  beforeEach(() => {
    jest.clearAllMocks();
    localStorage.clear();
    useChatStore.setState({
      activeConversationId: null,
      exportReadyByConversation: {},
      questionnaireReadyPayload: null,
    });

  describe('cache checks', () => {
    it('does not request export status if in-memory cache exists', () => {
      useChatStore.setState({
        activeConversationId: 'conv-123',
        exportReadyByConversation: {
          'conv-123': { assessmentId: 'a1', formats: ['pdf'], questionCount: 40 }
        },
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(mockWsClient.requestExportStatus).not.toHaveBeenCalled();
    });

    it('does not request export status if localStorage cache exists', () => {
      localStorage.setItem('export_conv-123', JSON.stringify({
        assessmentId: 'a1', formats: ['pdf'], questionCount: 40
      }));
      useChatStore.setState({
        activeConversationId: 'conv-123',
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(mockWsClient.requestExportStatus).not.toHaveBeenCalled();
      expect(useChatStore.getState().exportReadyByConversation['conv-123']).toBeDefined();
      expect(useChatStore.getState().questionnaireUIState).toBe('download');
    });

    it('removes corrupted localStorage data and continues to server', () => {
      localStorage.setItem('export_conv-123', 'not-json');
      mockWsClient.isConnected.mockReturnValue(true);
      useChatStore.setState({
        activeConversationId: 'conv-123',
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(localStorage.getItem('export_conv-123')).toBeNull();
      // Should continue to server request after cleanup
      expect(mockWsClient.requestExportStatus).toHaveBeenCalledWith('conv-123');
    });
  });

  describe('guards and short-circuits', () => {
    it('does not request if websocket not connected', () => {
      mockWsClient.isConnected.mockReturnValue(false);
      useChatStore.setState({
        activeConversationId: 'conv-123',
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(mockWsClient.requestExportStatus).not.toHaveBeenCalled();
    });

    it('requests export status even without questionnaireReadyPayload (fresh load scenario)', () => {
      // Critical: On fresh load, questionnaireReadyPayload is null but user may have
      // generated a questionnaire in a previous session. Server controls spam via 404.
      mockWsClient.isConnected.mockReturnValue(true);
      useChatStore.setState({
        activeConversationId: 'conv-123',
        // No questionnaireReadyPayload - simulates fresh page load
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(mockWsClient.requestExportStatus).toHaveBeenCalledWith('conv-123');
    });

    it('requests export status when connected and no cache', () => {
      mockWsClient.isConnected.mockReturnValue(true);
      useChatStore.setState({
        activeConversationId: 'conv-123',
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      expect(mockWsClient.requestExportStatus).toHaveBeenCalledWith('conv-123');
    });

    it('deduplicates concurrent export status requests', () => {
      mockWsClient.isConnected.mockReturnValue(true);
      useChatStore.setState({
        activeConversationId: 'conv-123',
      });

      const { rerender } = renderHook(() => useWebSocketEvents(mockWsClient as any));
      rerender();
      rerender();

      expect(mockWsClient.requestExportStatus).toHaveBeenCalledTimes(1);
    });
  });

  describe('export_status_not_found response', () => {
    it('clears pending request and stays in ready state', () => {
      let notFoundCallback: any;
      mockWsClient.onExportStatusNotFound.mockImplementation((cb) => {
        notFoundCallback = cb;
        return jest.fn();
      });

      useChatStore.setState({
        activeConversationId: 'conv-123',
      });
      renderHook(() => useWebSocketEvents(mockWsClient as any));

      act(() => {
        notFoundCallback({ conversationId: 'conv-123' });
      });

      // UI state should remain ready (no change)
      expect(useChatStore.getState().questionnaireUIState).not.toBe('download');
    });
  });

  describe('export_status_error response', () => {
    it('logs error and does not disrupt UI', () => {
      const consoleSpy = jest.spyOn(console, 'error').mockImplementation();
      let errorCallback: any;
      mockWsClient.onExportStatusError.mockImplementation((cb) => {
        errorCallback = cb;
        return jest.fn();
      });

      renderHook(() => useWebSocketEvents(mockWsClient as any));

      act(() => {
        errorCallback({ conversationId: 'conv-123', error: 'Auth failed' });
      });

      expect(consoleSpy).toHaveBeenCalledWith(
        '[useWebSocketEvents] Export status error:',
        'Auth failed'
      );
      consoleSpy.mockRestore();
    });
  });
});
```
