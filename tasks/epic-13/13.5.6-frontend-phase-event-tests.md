# Story 13.5.6 - Add Frontend Tests for Phase Events

## Status: Not Started

## Context
Frontend handles phase events (13.5.3). This story adds tests to verify correct event handling and state updates.

## Goal
Test that frontend correctly handles `generation_phase` events with conversation scoping and idempotency.

## Files to Create/Modify
- `apps/web/src/hooks/__tests__/useWebSocketEvents.test.ts` (update or create)

## Implementation

### Test Suite
```typescript
import { renderHook, act } from '@testing-library/react';
import { useWebSocketEvents } from '../useWebSocketEvents';
import { useChatStore, GENERATION_STEPS } from '@/stores/chatStore';
import { io as mockIo } from 'socket.io-client';

// Mock socket.io-client
jest.mock('socket.io-client');

describe('useWebSocketEvents - generation_phase handling', () => {
  let mockSocket: any;
  let socketEventHandlers: Record<string, Function>;
  const activeConversationId = 'active-convo-123';

  beforeEach(() => {
    socketEventHandlers = {};
    mockSocket = {
      on: jest.fn((event, handler) => {
        socketEventHandlers[event] = handler;
      }),
      off: jest.fn(),
      emit: jest.fn(),
      connected: true,
    };
    (mockIo as jest.Mock).mockReturnValue(mockSocket);

    // Reset store
    useChatStore.setState({
      activeConversationId,
      currentGenerationStep: -1,
      questionnaireUIState: 'hidden',
    });
  });

  const emitPhaseEvent = (phase: number, phaseId: string, conversationId = activeConversationId) => {
    act(() => {
      socketEventHandlers['generation_phase']?.({
        conversationId,
        phase,
        phaseId,
        timestamp: Date.now(),
      });
    });
  };

  describe('basic phase handling', () => {
    it('updates currentGenerationStep on phase event', () => {
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(1, 'generating');

      expect(useChatStore.getState().currentGenerationStep).toBe(1);
    });

    it('progresses through all phases in order', () => {
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(0, 'context');
      expect(useChatStore.getState().currentGenerationStep).toBe(0);

      emitPhaseEvent(1, 'generating');
      expect(useChatStore.getState().currentGenerationStep).toBe(1);

      emitPhaseEvent(2, 'validating');
      expect(useChatStore.getState().currentGenerationStep).toBe(2);

      emitPhaseEvent(3, 'saving');
      expect(useChatStore.getState().currentGenerationStep).toBe(3);
    });
  });

  describe('conversation scoping', () => {
    it('ignores phase events for other conversations', () => {
      useChatStore.setState({ currentGenerationStep: 0 });
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(2, 'validating', 'different-convo');

      // Should stay at 0, not advance to 2
      expect(useChatStore.getState().currentGenerationStep).toBe(0);
    });

    it('processes phase events for active conversation', () => {
      useChatStore.setState({ currentGenerationStep: 0 });
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(2, 'validating', activeConversationId);

      expect(useChatStore.getState().currentGenerationStep).toBe(2);
    });
  });

  describe('idempotency', () => {
    it('ignores out-of-order phase events (lower than current)', () => {
      useChatStore.setState({ currentGenerationStep: 2 });
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(1, 'generating');

      // Should stay at 2, not go back to 1
      expect(useChatStore.getState().currentGenerationStep).toBe(2);
    });

    it('ignores duplicate phase events (equal to current)', () => {
      useChatStore.setState({ currentGenerationStep: 2 });
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(2, 'validating');

      expect(useChatStore.getState().currentGenerationStep).toBe(2);
    });

    it('accepts forward phase events', () => {
      useChatStore.setState({ currentGenerationStep: 1 });
      renderHook(() => useWebSocketEvents());

      emitPhaseEvent(3, 'saving');

      expect(useChatStore.getState().currentGenerationStep).toBe(3);
    });
  });

  describe('export_ready handling', () => {
    it('marks stepper complete on export_ready', () => {
      useChatStore.setState({ currentGenerationStep: 3 });
      renderHook(() => useWebSocketEvents());

      act(() => {
        socketEventHandlers['export_ready']?.({
          conversationId: activeConversationId,
          formats: ['pdf', 'word', 'excel'],
          assessmentId: 'test-123',
        });
      });

      expect(useChatStore.getState().currentGenerationStep).toBe(GENERATION_STEPS.length);
    });

    it('ignores export_ready for other conversations', () => {
      useChatStore.setState({ currentGenerationStep: 3 });
      renderHook(() => useWebSocketEvents());

      act(() => {
        socketEventHandlers['export_ready']?.({
          conversationId: 'different-convo',
          formats: ['pdf'],
          assessmentId: 'test-123',
        });
      });

      // Should stay at 3, not mark complete
      expect(useChatStore.getState().currentGenerationStep).toBe(3);
    });
  });

  describe('error handling', () => {
    it('resets step to -1 on extraction_failed', () => {
      useChatStore.setState({ currentGenerationStep: 2 });
      renderHook(() => useWebSocketEvents());

      act(() => {
        socketEventHandlers['extraction_failed']?.({
          conversationId: activeConversationId,
          error: 'Generation failed',
        });
      });

      expect(useChatStore.getState().currentGenerationStep).toBe(-1);
    });
  });

  describe('cleanup', () => {
    it('removes listener on unmount', () => {
      const { unmount } = renderHook(() => useWebSocketEvents());

      unmount();

      expect(mockSocket.off).toHaveBeenCalledWith('generation_phase', expect.any(Function));
    });
  });
});
```
