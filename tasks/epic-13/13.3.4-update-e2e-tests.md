### 0. Test Configuration Prerequisites

#### 0.1. Enable Downloads in Playwright Context
**IMPORTANT:** Playwright needs `acceptDownloads: true` to capture download events:

```typescript
// In playwright.config.ts or test setup
import { defineConfig } from '@playwright/test';

export default defineConfig({
  use: {
    acceptDownloads: true,  // Required for download event handling
  },
});
```

#### 0.2. Stub Export API with page.route
**IMPORTANT:** For deterministic tests, stub the export API instead of hitting the real backend:

```typescript
// Add to beforeEach or test setup
await page.route('**/api/assessments/*/export/*', async (route) => {
  const url = route.request().url();
  const format = url.split('/').pop(); // 'pdf', 'word', 'excel'

  const contentTypes: Record<string, string> = {
    pdf: 'application/pdf',
    word: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
    excel: 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
  };

  const extensions: Record<string, string> = {
    pdf: 'pdf',
    word: 'docx',
    excel: 'xlsx',
  };

  await route.fulfill({
    status: 200,
    contentType: contentTypes[format!] || 'application/octet-stream',
    body: Buffer.from(`Mock ${format} content`),
    headers: {
      'Content-Disposition': `attachment; filename="questionnaire.${extensions[format!] || 'bin'}"`,
    },
  });
});
```

This ensures:
- Tests don't depend on backend availability
- Download file content is deterministic
- Tests run faster (no real file generation)

### 1. Find and Update Card Disappearance Assertions

Search for assertions like:

```typescript
// BEFORE: Expecting card to disappear
```

### 2. Add Multi-Download Test

```typescript
test('should allow multiple downloads without regenerating', async ({ page }) => {
  // Stub export API for deterministic tests
  await page.route('**/api/assessments/*/export/*', async (route) => {
    const format = route.request().url().split('/').pop();
    await route.fulfill({
      status: 200,
      contentType: format === 'pdf' ? 'application/pdf' : 'application/octet-stream',
      body: Buffer.from(`Mock ${format} content`),
    });
  });

  // Setup: Navigate to conversation with completed questionnaire
  await page.goto('/chat/conv-123');

  // Wait for download card to appear
  await expect(page.locator('[data-testid="questionnaire-card-download"]')).toBeVisible();

  // Download PDF (acceptDownloads: true must be set in config)
  const downloadPromise1 = page.waitForEvent('download');
  await page.locator('[data-testid="download-pdf-btn"]').click();
  const download1 = await downloadPromise1;
  expect(download1.suggestedFilename()).toContain('.pdf');

  // Verify card still visible
  await expect(page.locator('[data-testid="questionnaire-card-download"]')).toBeVisible();
  await expect(page.locator('[data-testid="download-word-btn"]')).toBeVisible();

  // Download Word
  const downloadPromise2 = page.waitForEvent('download');
  await page.locator('[data-testid="download-word-btn"]').click();
  const download2 = await downloadPromise2;
  expect(download2.suggestedFilename()).toContain('.docx');

  // Verify card STILL visible
  await expect(page.locator('[data-testid="questionnaire-card-download"]')).toBeVisible();
});
```

### 3. Add Page Reload Persistence Test

```typescript
test('should restore download state after page reload', async ({ page }) => {
  // Stub export API for deterministic tests
  await page.route('**/api/assessments/*/export/*', async (route) => {
    await route.fulfill({
      status: 200,
      contentType: 'application/pdf',
      body: Buffer.from('Mock PDF content'),
    });
  });

  // Setup: Navigate to conversation with completed questionnaire
  await page.goto('/chat/conv-123');

  // Wait for download card
  await expect(page.locator('[data-testid="questionnaire-card-download"]')).toBeVisible();

  // Download to confirm export state is saved (acceptDownloads: true required)
  const downloadPromise = page.waitForEvent('download');
  await page.locator('[data-testid="download-pdf-btn"]').click();
  await downloadPromise;

  // Reload page (clears in-memory state, should restore from localStorage)
  await page.reload();

  // Re-stub after reload (page.route needs to be re-applied)
  await page.route('**/api/assessments/*/export/*', async (route) => {
    await route.fulfill({
      status: 200,
      contentType: 'application/pdf',
      body: Buffer.from('Mock PDF content'),
    });
  });

  // Verify download card restored from localStorage
  await expect(page.locator('[data-testid="questionnaire-card-download"]')).toBeVisible();
  await expect(page.locator('[data-testid="download-pdf-btn"]')).toBeVisible();
});
```
