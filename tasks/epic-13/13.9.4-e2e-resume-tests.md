# Story 13.9.4 - E2E: Resume and Download Without Regeneration

## Status: Not Started

## Context
End-to-end tests to verify the complete resume flow works: user generates, leaves, returns, sees download buttons without regenerating.

**Note:** Regenerate tests removed (Story 13.9.3 deferred). Tests focus on persist/resume flow only.

## Goal
Add Playwright E2E tests for the export resume flow.

## Test Setup Requirements

### Test User
Tests require a seeded test user. Document in test setup:
```typescript
// Test user credentials (must exist in DB)
const TEST_USER = {
  email: 'e2e-test@guardian.local',
  password: 'E2ETestPassword123!',
};
```

### Test Isolation
**IMPORTANT:** Each test must be isolated to prevent cross-test pollution:
- Use unique conversation IDs or vendor names per test (e.g., `VendorName-${Date.now()}`)
- Reset localStorage in beforeEach (not just clear in test body)
- Consider DB fixture reset between test runs (CI)

### No networkidle
**IMPORTANT:** Do NOT use `page.waitForLoadState('networkidle')`. Use concrete element assertions instead:
- `await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 })`

### `apps/web/e2e/export-resume.spec.ts`

```typescript
import { test, expect } from '@playwright/test';

// Test user credentials (seeded in DB via fixture/migration)
const TEST_USER = {
  email: 'e2e-test@guardian.local',
  password: 'E2ETestPassword123!',
};

test.describe('Export Resume Flow (Story 13.9.4)', () => {

  test.beforeEach(async ({ page }) => {
    // Clear localStorage before each test for isolation
    await page.goto('/');
    await page.evaluate(() => localStorage.clear());

    // Login
    await page.goto('/login');
    await page.fill('[data-testid="email-input"]', TEST_USER.email);
    await page.fill('[data-testid="password-input"]', TEST_USER.password);
    await page.click('[data-testid="login-btn"]');

    // Wait for chat UI (not networkidle - use concrete element)
    await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 });
  });

  test('shows download buttons on resume without regeneration', async ({ page }) => {
    // Use unique vendor name for test isolation
    const vendorName = `ResumeTest-${Date.now()}`;

    // Step 1: Start a conversation and generate questionnaire
    await page.fill('[data-testid="chat-input"]', `Assess vendor ${vendorName} for their AI diagnostic tool`);
    await page.click('[data-testid="send-btn"]');

    // Wait for questionnaire ready prompt (concrete element, not networkidle)
    await expect(page.getByTestId('generate-questionnaire-btn')).toBeVisible({ timeout: 30000 });

    // Generate the questionnaire
    await page.click('[data-testid="generate-questionnaire-btn"]');

    // Wait for download buttons (generation complete)
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 90000 });

    // Verify we're in download state
    await expect(page.getByTestId('download-pdf-btn')).toBeVisible();
    await expect(page.getByTestId('download-excel-btn')).toBeVisible();

    // Step 2: Simulate "leaving" by clearing localStorage and refreshing
    await page.evaluate(() => localStorage.clear());
    await page.reload();

    // Wait for chat UI to be ready (not networkidle)
    await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 });

    // Step 3: Verify download buttons appear WITHOUT regenerating
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 15000 });

    // Verify generate button is NOT shown (we're in download state, not ready)
    await expect(page.getByTestId('generate-questionnaire-btn')).not.toBeVisible();
  });

  test('falls back to generate flow when no export exists', async ({ page }) => {
    // Use unique vendor name for test isolation
    const vendorName = `NewVendor-${Date.now()}`;

    // Create a new conversation (no previous export)
    await page.click('[data-testid="new-conversation-btn"]');

    // Send a message to trigger vendor context
    await page.fill('[data-testid="chat-input"]', `Assess vendor ${vendorName} for their platform`);
    await page.click('[data-testid="send-btn"]');

    // Wait for questionnaire ready prompt
    await expect(page.getByTestId('generate-questionnaire-btn')).toBeVisible({ timeout: 30000 });

    // Verify we're in ready state (generate button visible, download not)
    await expect(page.getByTestId('download-word-btn')).not.toBeVisible();
  });

  // NOTE: Regenerate tests removed - Story 13.9.3 deferred
  // Tests 'regenerate shows confirmation' and 'regenerate cancel keeps export'
  // will be added when 13.9.3 is implemented.

  test('download actually works after resume', async ({ page }) => {
    const vendorName = `DownloadTest-${Date.now()}`;

    // Generate questionnaire
    await page.fill('[data-testid="chat-input"]', `Assess vendor ${vendorName}`);
    await page.click('[data-testid="send-btn"]');
    await expect(page.getByTestId('generate-questionnaire-btn')).toBeVisible({ timeout: 30000 });
    await page.click('[data-testid="generate-questionnaire-btn"]');
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 90000 });

    // Clear localStorage and reload (simulate new session)
    await page.evaluate(() => localStorage.clear());
    await page.reload();
    await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 });

    // Wait for resume to complete
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 15000 });

    // Click download and verify it works
    const [download] = await Promise.all([
      page.waitForEvent('download'),
      page.click('[data-testid="download-word-btn"]'),
    ]);

    // Verify download started
    expect(download.suggestedFilename()).toMatch(/questionnaire.*\.docx$/);
  });

  test('multiple format downloads work after resume', async ({ page }) => {
    const vendorName = `MultiFormat-${Date.now()}`;

    // Setup: generate and resume
    await page.fill('[data-testid="chat-input"]', `Assess vendor ${vendorName}`);
    await page.click('[data-testid="send-btn"]');
    await expect(page.getByTestId('generate-questionnaire-btn')).toBeVisible({ timeout: 30000 });
    await page.click('[data-testid="generate-questionnaire-btn"]');
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 90000 });

    await page.evaluate(() => localStorage.clear());
    await page.reload();
    await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 });
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 15000 });

    // Download Word
    const [wordDownload] = await Promise.all([
      page.waitForEvent('download'),
      page.click('[data-testid="download-word-btn"]'),
    ]);
    expect(wordDownload.suggestedFilename()).toContain('.docx');

    // Download buttons should still be visible (durability)
    await expect(page.getByTestId('download-pdf-btn')).toBeVisible();

    // Download PDF
    const [pdfDownload] = await Promise.all([
      page.waitForEvent('download'),
      page.click('[data-testid="download-pdf-btn"]'),
    ]);
    expect(pdfDownload.suggestedFilename()).toContain('.pdf');
  });

  test('handles backend error gracefully on resume', async ({ page, context }) => {
    const vendorName = `ErrorTest-${Date.now()}`;

    // Generate questionnaire first
    await page.fill('[data-testid="chat-input"]', `Assess vendor ${vendorName}`);
    await page.click('[data-testid="send-btn"]');
    await expect(page.getByTestId('generate-questionnaire-btn')).toBeVisible({ timeout: 30000 });
    await page.click('[data-testid="generate-questionnaire-btn"]');
    await expect(page.getByTestId('download-word-btn')).toBeVisible({ timeout: 90000 });

    // Clear localStorage and reload
    await page.evaluate(() => localStorage.clear());
    await page.reload();
    await expect(page.getByTestId('chat-input')).toBeVisible({ timeout: 10000 });

    // Should not crash - chat UI should still be functional
    // Either download buttons appear (successful resume) or generate button (fallback)
    const hasDownload = await page.getByTestId('download-word-btn').isVisible({ timeout: 15000 }).catch(() => false);
    const hasGenerate = await page.getByTestId('generate-questionnaire-btn').isVisible().catch(() => false);

    // One of these should be true - UI should recover to a valid state
    expect(hasDownload || hasGenerate || await page.getByTestId('chat-input').isVisible()).toBe(true);
  });
});
```
